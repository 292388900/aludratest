45  @Component(role = AludraServiceManager.class, instantiationStrategy = "singleton")

54      private List<ServiceWrapper> serviceWrappers;

57      public AludraServiceManagerImpl() {
58          serviceRegistry = new AludraServiceRegistry();
59          new DefaultConfigurator().configure(serviceRegistry);
60      }

62      private <T> ComponentDescriptor<T> createComponentDescriptor(Class<T> iface, String instanceName, ClassRealm realm)
63              throws ClassNotFoundException {
64          ComponentDescriptor<T> cd = new ComponentDescriptor<T>();
65          cd.setRole(iface.getName());
66          cd.setRoleClass(iface);
67          if (instanceName != null) {
68              cd.setRoleHint(instanceName);
69          }
70  
71          String implClass = serviceRegistry.getImplementationClassName(iface.getName());
72          if (implClass == null) {
73              throw new AutomationException("No implementor class configured for interface " + iface.getName());
74          }
75          cd.setImplementation(implClass);
76          cd.setInstantiationStrategy(isSingleton(iface) ? "singleton" : "per-lookup");
77          cd.setRealm(realm);
78  
79          // lookup Requirements in class and parent classes
80          Class<?> clazz = realm.loadClass(implClass);
81          while (clazz != null && clazz != Object.class) {
82              for (Field f : clazz.getDeclaredFields()) {
83                  Requirement req = f.getAnnotation(Requirement.class);
84                  if (req != null) {
85                      cd.addRequirement(createComponentRequirement(f, req));
86                  }
87              }
88              clazz = clazz.getSuperclass();
89          }
90  
91          return cd;
92      }
93  
94      private ComponentRequirement createComponentRequirement(Field field, Requirement requirement) {
95          ComponentRequirement cr = new ComponentRequirement();
96          cr.setFieldName(field.getName());
97          cr.setOptional(requirement.optional());
98          if (requirement.role() != null && requirement.role() != Object.class) {
99              cr.setRole(requirement.role().getName());
100         }
101         else {
102             cr.setRole(field.getType().getName());
103         }
104         if (requirement.hint() != null) {
105             cr.setRoleHint(requirement.hint());
106         }
107 
108         return cr;
109     }

111     @Override
112     public <T extends AludraService> T createAndConfigureService(ComponentId<T> serviceId, AludraContext context, boolean wrap) {
113         // get service details
114         Class<T> serviceInterface = serviceId.getInterfaceClass();
115         String instanceName = serviceId.getInstanceName();
116 
117         // check preconditions
118         if (serviceInterface == null) {
119             throw new IllegalArgumentException("no serviceInterface specified");
120         }
121         if (!serviceInterface.isInterface()) {
122             throw new IllegalArgumentException(serviceInterface.getName() + " is not an interface");
123         }
124 
125         AludraServiceContext serviceContext = new AludraServiceContextImpl(context, instanceName);
126 
127         // instantiate and initialize service
128         try {
129             T service = newImplementorInstance(serviceInterface, serviceId.getInstanceName());
130             service.init(serviceContext);
131 
132             // wrap service with registered wrappers
133             if (serviceWrappers != null && wrap) {
134                 for (ServiceWrapper wrapper : serviceWrappers) {
135                     service = wrapper.wrap(service, serviceId, context);
136                 }
137             }
138 
139             return service;
140         }
141         catch (Exception e) {
142             if (e instanceof AludraTestException) {
143                 throw (AludraTestException) e;
144             }
145             else if (e instanceof ConfigurationException) {
146                 throw new AutomationException("Configuration exception when initializing " + serviceId, e);
147             }
148             else {
149                 throw new TechnicalException("Error initializing " + serviceId, e);
150             }
151         }
152     }
153 
154     @Override
155     public <T> T newImplementorInstance(Class<T> iface) {
156         return newImplementorInstance(iface, null);
157     }
158 
159     @Override
160     public void removeSingleton(Class<?> iface) {
161         Object o;
162         try {
163             o = container.lookup(iface);
164             // release calls close() if available
165             container.release(o);
166         }
167         catch (ComponentLookupException e) {
168             // ignore - component not found
169         }
170         catch (ComponentLifecycleException e) {
171             throw new TechnicalException("Exception when closing singleton of class " + iface.getName(), e);
172         }
173     }
174 
175     private boolean isSingleton(Class<?> iface) {
176         // check if interface class is marked as singleton; if so, check singleton map
177         if (iface.isAnnotationPresent(InternalComponent.class)) {
178             InternalComponent ic = iface.getAnnotation(InternalComponent.class);
179             return ic.singleton();
180         }
181         return false;
182     }
183 
184     private <T> T newImplementorInstance(Class<T> iface, String instanceName) {
185         String implementorClassName = serviceRegistry.getImplementationClassName(iface.getName());
186 
187         // use Plexus container for lookup
188         synchronized (container) {
189             if ((implementorClassName == null || implementorClassName.trim().isEmpty())
190                     && container.getComponentDescriptor(iface.getName(), instanceName) == null) {
191                 // if Plexus also does not know it, exception
192                 throw new AutomationException("No implementor class configured for interface " + iface.getName());
193             }
194 
195             if (container.getComponentDescriptor(iface.getName(), instanceName) == null) {
196                 try {
197                     ClassRealm realm = container.getLookupRealm();
198                     if (realm == null) {
199                         realm = container.getContainerRealm();
200                     }
201                     container.addComponentDescriptor(createComponentDescriptor(iface, instanceName, realm));
202                 }
203                 catch (CycleDetectedInComponentGraphException e) {
204                     throw new TechnicalException("Unexpected IoC error when instantiating object", e);
205                 }
206                 catch (ClassNotFoundException e) {
207                     throw new AutomationException("Could not find implementation class", e);
208                 }
209             }
210 
211             try {
212                 if (instanceName == null) {
213                     return container.lookup(iface);
214                 }
215                 else {
216                     return container.lookup(iface, instanceName);
217                 }
218             }
219             catch (ComponentLookupException e) {
220                 throw new TechnicalException("Could not lookup implementation for class " + iface.getName(), e);
221             }
222         }
223     }
