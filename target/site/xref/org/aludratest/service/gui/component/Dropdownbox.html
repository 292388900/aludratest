16  package org.aludratest.service.gui.component;
17  
18  import org.aludratest.service.locator.option.IndexLocator;
19  import org.aludratest.service.locator.option.LabelLocator;
20  import org.aludratest.service.locator.option.OptionLocator;
21  import org.databene.commons.Validator;
22  
23  /** Represents a dropdownbox in a GUI.
24   * @author Joerg Langnickel
25   * @author Volker Bergmann */
26  public interface Dropdownbox extends Element<Dropdownbox> {
27  
28      /** Selects an entry of this Dropdownbox by its label. If the label is <code>null</code> or marked as null, the operation is
29       * not executed. It the label is marked as empty, it will be replaced with "".
30       * 
31       * @param label Label of the entry to select. */
32      public void selectEntry(String label);
33  
34      /** Selects an entry of a Dropdownbox by its locator, which must be either a {@link LabelLocator} or an {@link IndexLocator}.
35       * 
36       * @param optionLocator is a LabelLocator or IndexLocator for identifying the entry to select. If <code>null</code>, no action
37       *            is performed. */
38      public void selectEntry(OptionLocator optionLocator);
39  
40      /** Reads the selected value and returns it as a String.
41       * 
42       * @return the <b>value</b> of the selected entry */
43      public String getSelectedEntry();
44  
45      /** Reads the selected label and returns it as a String.
46       * 
47       * @return the <b>label</b> of the selected entry */
48      public String getSelectedLabel();
49  
50      /** Verifies that the passed label is selected in this Dropdownbox. If the label is null or marked as null the operation is not
51       * executed. If the label is marked as empty it will be replaced with ""
52       * 
53       * @param label Label to check the currently selected label against. */
54      public void assertIsSelected(String label);
55  
56      /** Verifies that the passed label is <b>not</b> selected in this Dropdownbox. If the label is null or marked as null the
57       * operation is not executed. If the label is marked as empty it will be replaced with ""
58       * 
59       * @param label Label to check the currently selected label against. */
60      public void assertTextNotEquals(String label);
61  
62      /** Verifies that the passed text is part of the currently selected label in this Dropdownbox. If the label is null or marked
63       * as null the operation is not executed.
64       * 
65       * @param label Label to check the currently selected label against. */
66      public void assertSelectedContains(String label);
67  
68      /** Verifies that the passed label is selected in this Dropdownbox. If the label is null or marked as null the operation is not
69       * executed. If the label is marked as empty it will be replaced with "". Differences in case, leading or trailing whitespace
70       * are ignored.
71       * 
72       * @param label Label to check the currently selected label against. */
73      public void assertSelectedIgnoreCaseTrimmed(String label);
74  
75      /** Asserts that the selection matches the provided {@link Validator}.
76       * @param validator the validator to apply for verification */
77      public void assertSelectionMatches(Validator<String> validator);
78  
79      /** Verifies that this Dropdownbox has the passed through values (and only these). If only one value is passed and it is
80       * <code>null</code> or marked as <code>null</code>, the operation will not be executed.
81       * 
82       * @param values Values to check the values of this Dropdownbox against. */
83      public void assertHasValues(String... values);
84  
85      /** Checks if this Dropdownbox has the expected labels.
86       * 
87       * The check fails if:
88       * <ul>
89       * <li>order of the Labels is wrong</li>
90       * <li>not every Label is mentioned</li>
91       * <li>a wrong Label is mentioned</li>
92       * </ul>
93       * If only one label is passed and it is <code>null</code> or marked as <code>null</code>, the operation will not be executed.
94       * @param labels which should be checked */
95      public void assertHasLabels(String... labels);
96  
97      /** Verifies that this Dropdownbox has the given labels (and possibly more than these). If only one value is passed and it is
98       * <code>null</code> or marked as <code>null</code>, the operation will not be executed.
99       * 
100      * @param labels Labels to check the labels of this Dropdownbox against. */
101     public void assertContainsLabels(String... labels);
102 
103     /** Checks if this Dropdownbox has the given labels (and possibly more than these).
104      * 
105      * @param labels Labels to check the labels of this Dropdownbox against.
106      * 
107      * @return <code>true</code> if the checkbox contains the given labels, <code>false</code> otherwise. */
108     public boolean checkContainsLabels(String... labels);
109 
110     /** Checks if this Dropdownbox has ONLY the given labels - in the order as speficied.
111      * 
112      * @param labels Labels to check the labels of this Dropdownbox against.
113      * 
114      * @return <code>true</code> if the checkbox only contains the given labels, <code>false</code> otherwise. */
115     public boolean checkEqualsLabels(String... labels);
116 
117 }
