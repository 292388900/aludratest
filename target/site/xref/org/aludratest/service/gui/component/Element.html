16  package org.aludratest.service.gui.component;
17  
18  import org.aludratest.service.gui.AludraGUI;
19  import org.aludratest.service.locator.element.GUIElementLocator;
20  
21  /** Parent interface for GUI Elements e.g. Button.
22   * @author Joerg Langnickel
23   * @author Volker Bergmann
24   * @param <E> Type of the concrete element, to be used by subclasses. */
25  public interface Element<E extends Element<E>> extends GUIComponent {
26  
27      /** @return the element's locator */
28      public GUIElementLocator getLocator();
29  
30      /** Activates the <i>waiting until task completion</i> feature on the underlying aludraGUI with the default value for the task
31       * completion timeout.
32       * @return This object. */
33      public E waitingUntilTaskCompletion();
34  
35      /** Sets the {@link #taskCompletionTimeout} to the specified value.
36       * @param waitTime the wait time to apply. A negative number disables the wait feature, 0 implicitly activates the
37       *            {@link AludraGUI}'s default, a positive value is used as explicit timeout.
38       * @return This object. */
39      public E waitingUntilTaskCompletion(int waitTime);
40  
41      /** Asserts that the element is present */
42      public void assertPresent();
43  
44      /** Asserts that the element is not present */
45      public void assertNotPresent();
46  
47      /** Asserts that the element is visible */
48      public void assertVisible();
49  
50      /** Asserts that this element is enabled, i.e. does not have an active "disabled" state. */
51      public void assertEnabled();
52  
53      /** Asserts that this element is not enabled, i.e. has an active "disabled" state. */
54      public void assertNotEnabled();
55  
56      /** Asserts that the element has the focus. */
57      public void assertFocus();
58  
59      /** Sets the focus on this element */
60      public void focus();
61  
62      /** Double clicks the element. */
63      public void doubleClick();
64  
65      /** Single-clicks the element. */
66      public void click();
67  
68      /**
69       * Selectable click- clicks only when provided string is not null, not marked as null and provided string is "true"
70       * @param click -String
71       */
72      public void click(String click);
73  
74      /** Checks if the specified element is somewhere on the page within the standard timeout.
75       * 
76       * @return <code>true</code> if the element was found present somewhere on the page within the standard timeout,
77       *         <code>false</code> otherwise. */
78      public boolean isPresent();
79  
80      /**
81       * Checks if the specified element is somewhere on the page and in foreground
82       * within the standard timeout.
83       * 
84       * @return <code>true</code> if the element was found present somewhere on the page and in foreground within the standard timeout,
85       *         <code>false</code> otherwise. */
86      public boolean isPresentAndInForeground();
87  
88      /** Checks if the specified element is somewhere on the page within a given timeout
89       * 
90       * @param timeout max time to wait for the element to become present.
91       * @return <code>true</code> if the element was found present somewhere on the page within the given timeout,
92       *         <code>false</code> otherwise. */
93      public boolean isPresent(long timeout);
94  
95      /** Checks if the specified element is nowhere on the page within the standard timeout.
96       * 
97       * @return <code>true</code> if the element was <b>not</b> found present somewhere on the page within the standard timeout,
98       *         <code>false</code> otherwise. */
99      public boolean isNotPresent();
100 
101     /** Checks if the specified element is nowhere on the page within the standard timeout.
102      * 
103      * @param timeout time to check for the element not being present.
104      * 
105      * @return <code>true</code> if the element was <b>not</b> found present somewhere on the page within the given timeout,
106      *         <code>false</code> otherwise. */
107     public boolean isNotPresent(long timeout);
108 
109     /** Checks if the specified element is visible within the standard timeout.
110      * 
111      * @return <code>true</code> if the element is visible, <code>false</code> otherwise. */
112     public boolean isVisible();
113 
114     /** Checks if the specified element is visible within the given timeout.
115      * 
116      * @param timeout Max time to wait for the element to become visible.
117      * 
118      * @return <code>true</code> if the element was found visible during the given timeout, <code>false</code> otherwise. */
119     public boolean isVisible(long timeout);
120 
121     /** Checks whether this input element is enabled.
122      * 
123      * @return <code>true</code> if the input element is enabled, <code>false</code> otherwise. */
124     public boolean isEnabled();
125 
126     /** Checks whether this input element is enabled.
127      * 
128      * @param timeout Max time to wait for this input element to become enabled.
129      * 
130      * @return <code>true</code> if the input element is enabled, <code>false</code> otherwise. */
131     public boolean isEnabled(long timeout);
132 
133 }
