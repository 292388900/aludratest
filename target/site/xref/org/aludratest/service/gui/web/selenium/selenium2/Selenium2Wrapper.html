135     public Selenium2Wrapper(SeleniumWrapperConfiguration configuration, SeleniumResourceService resourceService) {
136         try {
137             this.configuration = configuration;
138             this.resourceService = resourceService;
139             // TODO support HTTPS
140             if ("http".equals(configuration.getUrlOfAutAsUrl().getProtocol()) && configuration.isUsingLocalProxy()) {
141                 this.proxy = getProxyPool().acquire();
142                 this.proxy.start();
143             }
144             this.usedSeleniumHost = resourceService.acquire();
145             this.seleniumUrl = new URL(usedSeleniumHost + "/wd/hub");
146             this.driver = newDriver();
147             this.driver.manage().timeouts().pageLoadTimeout(configuration.getTimeout(), TimeUnit.MILLISECONDS);
148             this.locatorSupport = new LocatorSupport(this.driver, configuration);
149         } catch (Exception e) {
150             LOGGER.error("Error initializing Selenium 2", e);
151             String host = usedSeleniumHost;
152             forceCloseApplicationUnderTest();
153             throw new TechnicalException(e.getMessage() + ". Used Host = " + host, e);
154         }
155     }
156 
157     private synchronized ProxyPool getProxyPool() {
158         if (proxyPool == null && "http".equals(configuration.getUrlOfAutAsUrl().getProtocol())
159                 && configuration.isUsingLocalProxy()) { // TODO support HTTPS
160             URL url = configuration.getUrlOfAutAsUrl();
161             proxyPool = new ProxyPool(url.getHost(), url.getPort(), configuration.getMinProxyPort(),
162                     resourceService.getHostCount());
163         }
164         return proxyPool;
165     }

167     private WebDriver newDriver() {
168         try {
169             String driverName = configuration.getDriverName();
170             Selenium2Driver driverEnum = Selenium2Driver.valueOf(driverName);
171 
172             if (configuration.isUsingRemoteDriver()) {
173                 return driverEnum.newRemoteDriver(seleniumUrl, configuration.getBrowserArguments());
174             }
175             else {
176                 return driverEnum.newLocalDriver();
177             }
178         }
179         catch (Exception e) {
180             throw new TechnicalException("WebDriver creation failed: ", e);
181         }
182     }

194     private void forceCloseApplicationUnderTest() {
195         if (this.proxy != null) {
196             this.proxy.stop();
197             getProxyPool().release(this.proxy);
198             this.proxy = null;
199         }
200         if (this.driver != null) {
201             for (String id : getAllWindowIDs()) {
202                 try {
203                     selectWindowByTechnicalName(id);
204                     close();
205                 }
206                 catch (Exception e) {
207                     // ignore during close
208                 }
209             }
210             quit();
211             this.driver = null;
212         }
213         if (this.usedSeleniumHost != null) {
214             resourceService.release(this.usedSeleniumHost);
215             this.usedSeleniumHost = null;
216         }
217     }

242     @SuppressWarnings("unchecked")
243     private WebElement doBeforeDelegate(GUIElementLocator locator, boolean visible, boolean actionPending, boolean enabled) {
244         if (actionPending) {
245             waitUntilNotBusy();
246         }
247         MixedElementCondition condition = new MixedElementCondition(locator, locatorSupport, visible, enabled);
248         try {
249             WebElement element = waitFor(condition, configuration.getTimeout());
250             highlight(locator);
251             return element;
252         }
253         catch (TimeoutException e) {
254             throw new AutomationException(condition.getMessage());
255         }
256     }

259         if (taskCompletionTimeout >= 0) {
260             int timeout = (taskCompletionTimeout == 0 ? configuration.getTaskCompletionTimeout() : taskCompletionTimeout);
261             TaskCompletionUtil.waitForActivityAndCompletion(systemConnector, failureMessage, configuration.getTaskStartTimeout(),
262                     timeout, configuration.getTaskPollingInterval());
263         }
264     }

370     public void selectWindowImmediately(WindowLocator locator) {
371         LOGGER.debug("selectWindowImmediately({})", locator);
372         if (locator instanceof TitleLocator) {
373             String requestedTitle = locator.toString();
374 
375             // performance tweak: if the current window is the requested one, return immediately
376             try {
377                 LOGGER.debug("driver.getTitle()");
378                 if (requestedTitle.equals(driver.getTitle())) {
379                     return;
380                 }
381             }
382             catch (WebDriverException e) {
383                 // this happens, when trying to call getTitle()
384                 // on a driver which has just close()d the recent window.
385                 // In this case, I fall back to iterating all windows below
386             }
387 
388             removeHighlight();
389 
390             // iterate all windows and return the one with the desired title
391             StringBuilder sb = new StringBuilder();
392             for (String handle : getWindowHandles()) {
393                 try {
394                     LOGGER.debug("driver.switchTo().window()", locator);
395                     String title = driver.switchTo().window(handle).getTitle();
396                     if (sb.length() > 0) {
397                         sb.append(", ");
398                     }
399                     sb.append("\"").append(title).append("\"");
400                     if (requestedTitle.equals(title)) {
401                         return;
402                     }
403                 }
404                 catch (WebDriverException e) {
405                     // ignore; window has been removed in the meantime
406                 }
407             }
408 
409             // if the window has not been found, throw an ElementNotFoundException
410             throw new AutomationException("Element not found. Available window titles are: " + sb.toString());
411         }
412         else {
413             throw ServiceUtil.newUnsupportedLocatorException(locator);
414         }
415     }

424     public Map<String, String> getAllWindowHandlesAndTitles() {
425         LOGGER.debug("getAllWindowHandlesAndTitles()");
426         Map<String, String> handlesAndTitles = new HashMap<String, String>();
427         // store handle and title of current window
428         String initialWindowHandle;
429         try {
430             LOGGER.debug("driver.getWindowHandle()");
431             initialWindowHandle = driver.getWindowHandle();
432         }
433         catch (NoSuchWindowException e) {
434             // fallback to next best window - current window has just been closed!
435             LOGGER.debug("driver.getWindowHandles()");
436             Set<String> handles = driver.getWindowHandles();
437             if (handles.isEmpty()) {
438                 return Collections.emptyMap();
439             }
440 
441             initialWindowHandle = handles.iterator().next();
442             LOGGER.debug("driver.switchTo().window(initialWindowHandle)");
443             driver.switchTo().window(initialWindowHandle);
444         }
445 
446         try {
447             String title = getTitle();
448             handlesAndTitles.put(initialWindowHandle, title);
449         }
450         catch (WebDriverException e) {
451             // ignore current window
452             LOGGER.warn("Could not determine title of current window. Assuming window has just been closed.");
453         }
454 
455         // iterate all other windows by handle and get their titles
456         String currentHandle = initialWindowHandle;
457         Set<String> handles = getWindowHandles();
458         for (String handle : handles) {
459             if (!handle.equals(initialWindowHandle)) {
460                 LOGGER.debug("Switching to window with handle {}", handle);
461                 try {
462                     LOGGER.debug("driver.switchTo().window(handle)");
463                     driver.switchTo().window(handle);
464                     currentHandle = handle;
465                     String title;
466                     handlesAndTitles.put(handle, title = driver.getTitle());
467                     LOGGER.debug("Window with handle {} has title '{}'", handle, title);
468                 }
469                 catch (NoSuchWindowException e) {
470                     // ignore this window
471                 }
472                 catch (WebDriverException e) {
473                     // ignore this window
474                     LOGGER.warn("WebDriverException when querying window with handle " + handle
475                             + ". Assuming window has just been closed.");
476                 }
477             }
478         }
479         // switch back to the original window
480         if (!currentHandle.equals(initialWindowHandle)) {
481             try {
482                 LOGGER.debug("driver.switchTo().window(initialWindowHandle)");
483                 driver.switchTo().window(initialWindowHandle);
484             }
485             catch (WebDriverException e) {
486                 // selenium could now be on an unexpected window
487                 LOGGER.warn("Could not switch back to initial window after window iteration. Active window is now unspecified.");
488             }
489         }
490         return handlesAndTitles;
491     }

558     private Set<String> getWindowHandles() {
559         LOGGER.debug("getWindowHandles()");
560         // try up to three times when getting strange WebDriverExceptions
561         Callable<Set<String>> callable = new Callable<Set<String>>() {
562             @Override
563             public Set<String> call() throws Exception {
564                 try {
565                     Set<String> handles = driver.getWindowHandles();
566                     LOGGER.debug("getWindowHandles() -> {}", handles);
567                     return handles;
568                 }
569                 catch (Exception e) {
570                     // wait a little bit in case of exception, e.g. for browser window to close
571                     try {
572                         Thread.sleep(100);
573                     }
574                     catch (InterruptedException ie) {
575                     }
576                     throw e;
577                 }
578             }
579         };
580         try {
581             if (driver instanceof RemoteWebDriver) {
582                 // also reduce timeout for TCP connection, in case remote hangs
583                 ((AludraSeleniumHttpCommandExecutor) ((RemoteWebDriver) driver).getCommandExecutor()).setRequestTimeout(5000);
584             }
585             return RetryService.call(callable, WebDriverException.class, 2);
586         }
587         catch (Throwable t) {
588             LOGGER.error("Could not retrieve window handles", t);
589             return Collections.emptySet();
590         }
591         finally {
592             if (driver instanceof RemoteWebDriver) {
593                 // also reduce timeout for TCP connection, in case remote hangs
594                 ((AludraSeleniumHttpCommandExecutor) ((RemoteWebDriver) driver).getCommandExecutor()).setRequestTimeout(0);
595             }
596         }
597     }

656         LOGGER.debug("getTable({})", locator);
657         WebElement table = findElementImmediately(locator);
658         List<WebElement> tbodies = table.findElements(By.tagName("tbody"));
659         WebElement rowHolder = (tbodies.size() > 0 ? tbodies.get(0) : table);
660         List<WebElement> trs = rowHolder.findElements(By.tagName("tr"));
661         if (row >= trs.size()) {
662             throw new AutomationException("Table cell not found. " + "Requested row index " + row + " of " + trs.size()
663                     + " rows ");
664         }
665         List<WebElement> tds = trs.get(row).findElements(By.tagName("td"));
666         if (col >= tds.size()) {
667             throw new AutomationException("Table cell not found. " + "Requested column index " + col + " of " + tds.size()
668                     + " cells ");
669         }
670         return tds.get(col).getText();
671     }

897     private String captureScreenshotToString() {
898         // use Selenium1 interface to capture full screen
899         String url = seleniumUrl.toString();
900         Pattern p = Pattern.compile("(http(s)?://.+)/wd/hub(/?)");
901         Matcher matcher = p.matcher(url);
902         if (matcher.matches()) {
903             String screenshotUrl = matcher.group(1);
904             screenshotUrl += (screenshotUrl.endsWith("/") ? "" : "/") + "selenium-server/driver/?cmd=captureScreenshotToString";
905             InputStream in = null;
906             try {
907                 in = new URL(screenshotUrl).openStream();
908                 in.read(new byte[3]); // read away "OK,"
909                 ByteArrayOutputStream baos = new ByteArrayOutputStream();
910                 IOUtils.copy(in, baos);
911                 return new String(baos.toByteArray(), "UTF-8");
912             }
913             catch (IOException e) {
914                 // OK, fallthrough to Selenium 2 method
915             }
916             finally {
917                 IOUtils.closeQuietly(in);
918             }
919         }
920 
921         WebDriver screenshotDriver;
922         if (RemoteWebDriver.class.isAssignableFrom(driver.getClass())) {
923             screenshotDriver = new Augmenter().augment(driver);
924         }
925         else {
926             screenshotDriver = driver;
927         }
928         if (screenshotDriver instanceof TakesScreenshot) {
929             TakesScreenshot tsDriver = (TakesScreenshot) screenshotDriver;
930             return tsDriver.getScreenshotAs(OutputType.BASE64);
931         }
932         else {
933             throw new UnsupportedOperationException(driver.getClass() + " does not implement TakeScreenshot");
934         }
935     }

946     private void removeHighlight() {
947         if (configuration.getHighlightCommands() && this.highlightedElement != null) {
948             // unwrap element if wrapped, for direct access
949             while (highlightedElement instanceof ElementWrapper) {
950                 highlightedElement = ((ElementWrapper) highlightedElement).getWrappedElement();
951             }
952             try {
953                 // first try via ID
954                 String id = highlightedElement.getAttribute("id");
955                 if (id != null) {
956                     executeScript("document.getElementById('" + id + "').className = document.getElementById('" + id
957                             + "').className.replace( /(?:^|\\s)selenium-highlight(?!\\S)/g , '' );");
958                 }
959                 else {
960                     // fallback via Selenium arguments infrastructure
961                     executeScript(
962                             "arguments[0].className = arguments[0].className.replace( /(?:^|\\s)selenium-highlight(?!\\S)/g , '' );",
963                             highlightedElement);
964                 }
965             }
966             catch (WebDriverException e) {
967                 LOGGER.trace("Highlight remove failed. ", e);
968             }
969         }
970         highlightedElement = null;
971     }

981             // add hidden element
982             executeScript("var hidden = document.createElement('div'); hidden.setAttribute('id', '__aludra_selenium_hidden'); hidden.setAttribute('style', 'display:none;'); document.getElementsByTagName('body')[0].appendChild(hidden);");
983         }

1011         LOGGER.debug("executeScript({})", script);
1012         // first unwrap all possibly wrapped arguments of type WebElement, or JavaScript/JSON will fail

1019         return ((JavascriptExecutor) driver).executeScript(script, arguments);

1044     private void setValue(WebElement element, String value) {
1045         LOGGER.debug("setValue(WebElement, {})", value);
1046         sendKeys(element, Keys.END);
1047         String text;
1048         int tryCounter = 3;
1049         while (tryCounter > 0 && !DataMarkerCheck.isNull(text = getValue(element))) {
1050             int length = text.length();
1051             String[] arr = new String[length];
1052             for (int i = 0; i < length; i++) {
1053                 arr[i] = "\b";
1054             }
1055             sendKeys(element, arr);
1056             tryCounter--;
1057         }
1058         if (tryCounter == 0) {
1059             throw new AutomationException("Could not clear input field. Maybe covered by other component?");
1060         }
1061         sendKeys(element, value);
1062 
1063         try {
1064             sendKeys(element, Keys.TAB);
1065         }
1066         catch (StaleElementReferenceException e) {
1067             // ignore; key could have caused page change
1068             LOGGER.debug("Could not fire change event for element because element is now stale.");
1069         }
1070     }

1108     // special handling of some Selenium exceptions ----------------------------
1109 
1110     private void handleSeleniumException(Throwable e) {
1111         String message = e.getMessage();
1112 
1113         // check if there is a WebDriverException
1114         WebDriverException wde = null;
1115         while (e != null) {
1116             if (e instanceof WebDriverException) {
1117                 wde = (WebDriverException) e;
1118                 break;
1119             }
1120             e = e.getCause();
1121         }
1122 
1123         if (wde != null) {
1124             if (message == null) {
1125                 message = wde.getMessage();
1126             }
1127             // "not clickable" exception
1128             Pattern p = Pattern.compile("(unknown error: )?(.* not clickable .*)");
1129             Matcher m;
1130             if (message != null && (m = p.matcher(message)).find() && m.start() == 0) {
1131                 throw new AutomationException(m.group(2), wde);
1132             }
1133 
1134             // NoSuchElementException
1135             if (wde instanceof NoSuchElementException) {
1136                 throw new AutomationException("Element not found", wde);
1137             }
1138 
1139             // check for a time out
1140             if (wde instanceof TimeoutException) {
1141                 throw new PerformanceFailure(wde.getMessage(), wde);
1142             }
1143 
1144             throw wde;
1145         }
1146 
1147 
1148         // otherwise, throw a technical exception
1149         throw new TechnicalException("Unknown exception when clicking element", e);
1150     }
