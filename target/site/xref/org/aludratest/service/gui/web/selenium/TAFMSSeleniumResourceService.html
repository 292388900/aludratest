61  /** Uses a TAFMS server to obtain Selenium resources. The TAFMS server deals with priorities among different users.
62   * 
63   * @author falbrech */
64  @ConfigProperties({
65      @ConfigProperty(name = "tafms.url", type = String.class, description = "The base URL to the TAFMS server providing the resources", defaultValue = "http://127.0.0.1:8080/tafms"),
66      @ConfigProperty(name = "tafms.user", type = String.class, description = "The user name to use for the TAFMS server", required = true),
67      @ConfigProperty(name = "tafms.password", type = String.class, description = "The password to use for the TAFMS server", required = true),
68      @ConfigProperty(name = "tafms.jobName", type = String.class, description = "A name to send to TAFMS as identifying job name. This is used for logging and statistics.", required = false),
69      @ConfigProperty(name = "tafms.niceLevel", type = int.class, description = "The nice level to use for this job (-20 to 19). This only affects the priority in relation to other running jobs by the same TAFMS user. The lower the value, the higher the priority.", defaultValue = "0") })
70  public class TAFMSSeleniumResourceService implements SeleniumResourceService, Configurable {
71  
72      private static final Logger LOG = LoggerFactory.getLogger(TAFMSSeleniumResourceService.class);
73  
74      @Requirement
75      private AludraTestConfig aludraConfig;
76  
77      private Preferences configuration;
78  
79      private Map<String, String> hostResourceIds = new ConcurrentHashMap<String, String>();
80  
81      @Override
82      public String acquire() {
83          // prepare a JSON query to the given TAFMS server
84          JSONObject query = new JSONObject();
85  
86          try {
87              query.put("resourceType", "selenium");
88              query.put("niceLevel", configuration.getIntValue("tafms.niceLevel", 0));
89              String jobName = configuration.getStringValue("tafms.jobName");
90              if (jobName != null && !"".equals(jobName)) {
91                  query.put("jobName", jobName);
92              }
93          }
94          catch (JSONException e) {
95          }
96  
97          // prepare authentication
98          BasicCredentialsProvider provider = new BasicCredentialsProvider();
99          provider.setCredentials(AuthScope.ANY, new UsernamePasswordCredentials(configuration.getStringValue("tafms.user"),
100                 configuration.getStringValue("tafms.password")));
101 
102         CloseableHttpClient client = HttpClientBuilder.create().setConnectionReuseStrategy(new NoConnectionReuseStrategy())
103                 .disableConnectionState().disableAutomaticRetries().setDefaultCredentialsProvider(provider).build();
104 
105         String message = null;
106         try {
107             boolean wait;
108 
109             // use preemptive authentication to avoid double connection count
110             AuthCache authCache = new BasicAuthCache();
111             // Generate BASIC scheme object and add it to the local auth cache
112             BasicScheme basicAuth = new BasicScheme();
113             URL url = new URL(getTafmsUrl());
114             HttpHost host = new HttpHost(url.getHost(), url.getPort() == -1 ? url.getDefaultPort() : url.getPort(),
115                     url.getProtocol());
116             authCache.put(host, basicAuth);
117 
118             // Add AuthCache to the execution context
119             BasicHttpContext localcontext = new BasicHttpContext();
120             localcontext.setAttribute(HttpClientContext.AUTH_CACHE, authCache);
121 
122             do {
123                 // send a POST request to resource URL
124                 HttpPost request = new HttpPost(getTafmsUrl() + "resource");
125 
126                 // attach query as JSON string data
127                 request.setEntity(new StringEntity(query.toString(), ContentType.APPLICATION_JSON));
128 
129                 CloseableHttpResponse response = null;
130 
131                 // fire request
132                 response = client.execute(request, localcontext);
133 
134                 try {
135                     if (response.getStatusLine() == null) {
136                         throw new ClientProtocolException("No HTTP status line transmitted");
137                     }
138 
139                     message = extractMessage(response);
140                     if (response.getStatusLine().getStatusCode() != HttpStatus.SC_OK) {
141                         LOG.error("Exception when querying TAFMS server for resource. HTTP Status: "
142                                 + response.getStatusLine().getStatusCode() + ", message: " + message);
143                         return null;
144                     }
145 
146                     JSONObject object = new JSONObject(message);
147                     if (object.has("errorMessage")) {
148                         LOG.error("TAFMS server reported an error: " + object.get("errorMessage"));
149                         return null;
150                     }
151 
152                     // continue wait?
153                     if (object.has("waiting") && object.getBoolean("waiting")) {
154                         wait = true;
155                         query.put("requestId", object.getString("requestId"));
156                     }
157                     else {
158                         JSONObject resource = object.optJSONObject("resource");
159                         if (resource == null) {
160                             LOG.error("TAFMS server response did not provide a resource. Message was: " + message);
161                             return null;
162                         }
163 
164                         String sUrl = resource.getString("url");
165                         hostResourceIds.put(sUrl, object.getString("requestId"));
166 
167                         return sUrl;
168                     }
169                 }
170                 finally {
171                     IOUtils.closeQuietly(response);
172                 }
173             }
174             while (wait);
175 
176             // should never come here
177             return null;
178         }
179         catch (ClientProtocolException e) {
180             LOG.error("Exception in HTTP transmission", e);
181             return null;
182         }
183         catch (IOException e) {
184             LOG.error("Exception in communication with TAFMS server", e);
185             return null;
186         }
187         catch (JSONException e) {
188             LOG.error("Invalid JSON received from TAFMS server." + (message != null ? " JSON message was: " + message : ""), e);
189             return null;
190         }
191         finally {
192             IOUtils.closeQuietly(client);
193         }
194     }
195 
196     @Override
197     public void release(String server) {
198         if (server == null) {
199             return;
200         }
201 
202         String resourceKey = hostResourceIds.remove(server);
203         if (resourceKey == null) {
204             return;
205         }
206 
207         CloseableHttpClient client = HttpClientBuilder.create().setConnectionReuseStrategy(new NoConnectionReuseStrategy())
208                 .disableConnectionState().disableAutomaticRetries().build();
209 
210         // send a DELETE request to resource URL
211         HttpDelete request = new HttpDelete(getTafmsUrl() + "resource/" + resourceKey);
212         CloseableHttpResponse response = null;
213 
214         try {
215             response = client.execute(request);
216         }
217         catch (IOException e) {
218             LOG.warn("Could not release TAFMS resource", e);
219         }
220         finally {
221             IOUtils.closeQuietly(response);
222             IOUtils.closeQuietly(client);
223         }
224     }

241     @Override
242     public void configure(Preferences preferences) {
243         int niceLevel = preferences.getIntValue("tafms.niceLevel", 0);
244         if (niceLevel < -20 || niceLevel > 19) {
245             throw new AutomationException("Illegal value for tafms.niceLevel: " + niceLevel
246                     + ". Value must be from -20 to +19, inclusive");
247         }
248 
249         String url = preferences.getStringValue("tafms.url");
250         try {
251             new URL(url);
252         }
253         catch (Exception e) {
254             throw new AutomationException("Illegal URL for tafms.url: " + url, e);
255         }
256 
257         String user = preferences.getStringValue("tafms.user");
258         if (user == null || "".equals(user)) {
259             throw new AutomationException("TAFMS user name is missing");
260         }
261         String password = preferences.getStringValue("tafms.password");
262         if (password == null || "".equals(password)) {
263             throw new AutomationException("TAFMS password is missing");
264         }
265 
266         configuration = preferences;
267     }
268 
269     private String getTafmsUrl() {
270         String url = configuration.getStringValue("tafms.url");
271         if (!url.endsWith("/")) {
272             url += "/";
273         }
274         return url;
275     }
276 
277     private String extractMessage(HttpResponse response) throws IOException {
278         if (response.getEntity() == null) {
279             return null;
280         }
281 
282         HttpEntity entity = response.getEntity();
283         InputStream in = entity.getContent();
284 
285         try {
286             ByteArrayOutputStream baos = new ByteArrayOutputStream();
287             IOUtils.copy(in, baos);
288             return new String(baos.toByteArray(), "UTF-8");
289         }
290         finally {
291             EntityUtils.consumeQuietly(entity);
292             IOUtils.closeQuietly(in);
293         }
294     }
