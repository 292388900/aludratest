55  /**
56   * Opens a server socket at the {@link #localPort}, receives HTTP requests,
57   * injects custom HTTP request headers 
58   * and forwards the request to the server {@link #realHost}.   
59   * @author Volker Bergmann
60   */
61  public class AuthenticatingHttpProxy {
62  
63      /** The logger of the class */
64      private static final Logger LOGGER = LoggerFactory.getLogger(AuthenticatingHttpProxy.class);
65  
66      /** The buffer size for the socket */
67      private static final int BUFSIZE = 8 * 1024;
68  
69      /** The String-formatted IP address of the host on which the process is running. */
70      private String localHost;
71  
72      /** The local port on which to listen for incoming connections. */
73      private int localPort;
74  
75      /** The host to which to forward the incoming requests. */
76      private HttpHost realHost;
77  
78      /** The custom HTTP header information to inject. */
79      private Map<String, String> customHeaders;
80  
81      /** A boolean flag for controlling the server's main listener loop. */
82      private boolean running;
83  
84      private ServerThread serverThread;
85  
86      /** Constructor with the central configuration settings. */
87      public AuthenticatingHttpProxy(int localPort, String targetHost, int targetPort) {
88          try {
89              this.localHost = InetAddress.getLocalHost().getHostAddress();
90              this.localPort = localPort;
91              this.realHost = new HttpHost(targetHost, targetPort);
92              this.customHeaders = new HashMap<String, String>();
93              this.running = false;
94              this.serverThread = null;
95          } catch (UnknownHostException e) {
96              throw new AutomationException("Error initializing " + getClass(), e);
97          }
98      }
99  
100     public boolean isRunning() {
101         return running;
102     }
103 
104     /** sets an 'Authorization' HTTP request header for BASIC authentication. */
105     public void setBasicAuthentication(String user, String password) {
106         if (user != null || password != null) {
107             String code = Base64Codec.encode((user + ":" + password).getBytes(DataUtil.UTF_8));
108             setCustomRequestHeader("Authorization", "Basic " + code);
109         }
110     }
111 
112     /** Sets the name-value pair of a custom HTTP request header to add to forwarded calls.
113      *  For each request header to be set, all incoming request headers of the same key 
114      *  are deleted, then the new value is set. */
115     public void setCustomRequestHeader(String key, String value) {
116         customHeaders.put(key, value);
117     }
118 
119     public String mapTargetToProxyUrl(String requestedUrlString) {
120         try {
121             URL requestedUrl = new URL(requestedUrlString);
122             int requestedPort = requestedUrl.getPort();
123             if (requestedPort == -1) {
124                 requestedPort = 80; // NOSONAR
125             }
126             if (requestedUrl.getHost().equals(realHost.getHostName()) && requestedPort == realHost.getPort()) {
127                 URL proxyUrl = new URL(requestedUrl.getProtocol(), localHost, localPort, requestedUrl.getFile());
128                 return proxyUrl.toString();
129             } else {
130                 return requestedUrlString;
131             }
132         } catch (MalformedURLException e) {
133             throw new RuntimeException(e);
134         }
135     }
136 
137     public void start() throws IOException {
138         serverThread = new ServerThread(localPort, realHost);
139         this.running = true;
140         serverThread.start();
141     }
142 
143     public void stop() {
144         this.running = false;
145         try {
146             if (this.serverThread != null) { // this only happens when start() has failed
147                 this.serverThread.serversocket.close();
148                 this.serverThread = null;
149             }
150         } catch (IOException e) {
151             LOGGER.error("Error closing server socket. ", e);
152         }
153     }
154 
155     class ServerThread extends Thread {
156 
157         private final HttpHost target;
158         private final HttpService httpService;
159         private final ServerSocket serversocket;
160 
161         public ServerThread(int port, HttpHost target) throws IOException {
162             this.target = target;
163             this.serversocket = new ServerSocket(port);
164             this.setDaemon(true);
165 
166             HttpRequestInterceptor authenticator = new HttpRequestInterceptor() {
167                 @Override
168                 public void process(HttpRequest request, HttpContext context) throws HttpException, IOException {
169                     for (Map.Entry<String, String> customHeader : customHeaders.entrySet()) {
170                         String key = customHeader.getKey();
171                         String value = customHeader.getValue();
172                         request.removeHeaders(key);
173                         if (value != null) {
174                             request.addHeader(key, value);
175                         }
176                     }
177                 }
178             };
179 
180             // Set up HTTP protocol processor for incoming connections
181             HttpProcessor inhttpproc = new ImmutableHttpProcessor(new HttpRequestInterceptor[] {
182                     //new RequestContent(),
183                     authenticator, new RequestHeaderOverwriter(HTTP.TARGET_HOST, target.toHostString()), new RequestConnControl(), new RequestExpectContinue(true) });
184 
185             // Set up HTTP protocol processor for outgoing connections
186             HttpProcessor outhttpproc = new ImmutableHttpProcessor(new HttpResponseInterceptor[] { new ResponseDate(), new ResponseContent(), new ResponseConnControl() });
187 
188             // Set up outgoing request executor
189             HttpRequestExecutor httpexecutor = new HttpRequestExecutor();
190 
191             // Set up incoming request handler
192             UriHttpRequestHandlerMapper reqistry = new UriHttpRequestHandlerMapper();
193             reqistry.register("*", new ProxyHandler(this.target, outhttpproc, httpexecutor));
194 
195             // Set up the HTTP service
196             this.httpService = new HttpService(inhttpproc, reqistry);
197         }
198 
199         @Override
200         public void run() {
201             LOGGER.info("Listening on " + localHost + ":" + localPort);
202             while (running && !Thread.interrupted()) {
203                 try {
204                     int bufsize = BUFSIZE;
205                     // Set up incoming HTTP connection
206                     Socket insocket = this.serversocket.accept();
207                     DefaultBHttpServerConnection inconn = new DefaultBHttpServerConnection(bufsize);
208                     LOGGER.debug("Incoming connection from " + insocket.getInetAddress());
209                     inconn.bind(insocket);
210 
211                     // Set up outgoing HTTP connection
212                     Socket outsocket = new Socket(this.target.getHostName(), this.target.getPort());
213                     DefaultBHttpClientConnection outconn = new DefaultBHttpClientConnection(bufsize);
214                     outconn.bind(outsocket);
215                     LOGGER.debug("Outgoing connection to " + outsocket.getInetAddress());
216 
217                     // Start worker thread
218                     Thread t = new RequestProcessorThread(AuthenticatingHttpProxy.this, this.httpService, inconn, outconn);
219                     t.setDaemon(true);
220                     t.start();
221                 } catch (InterruptedIOException e) {
222                     break;
223                 } catch (SocketException e) {
224                     if (running) {
225                         LOGGER.error("Socket error: " + e.getMessage(), e);
226                     }
227                     break;
228                 } catch (IOException e) {
229                     LOGGER.error("I/O error initialising connection thread: " + e.getMessage(), e);
230                     break;
231                 }
232             }
233             if (this.serversocket != null) {
234                 try {
235                     this.serversocket.close();
236                 } catch (IOException e) {
237                     LOGGER.error("Error closing server socket. ", e);
238                 }
239             }
240         }
241     }
