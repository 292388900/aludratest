16  package org.aludratest.service.gui.web.selenium.selenium2;
17  
18  import java.net.URL;
19  import java.util.HashMap;
20  import java.util.Map;
21  
22  import org.databene.commons.BeanUtil;
23  import org.openqa.selenium.WebDriver;
24  import org.openqa.selenium.chrome.ChromeDriver;
25  import org.openqa.selenium.chrome.ChromeOptions;
26  import org.openqa.selenium.firefox.FirefoxDriver;
27  import org.openqa.selenium.htmlunit.HtmlUnitDriver;
28  import org.openqa.selenium.ie.InternetExplorerDriver;
29  import org.openqa.selenium.remote.BrowserType;
30  import org.openqa.selenium.remote.DesiredCapabilities;
31  import org.openqa.selenium.remote.LocalFileDetector;
32  import org.openqa.selenium.remote.RemoteWebDriver;
33  import org.openqa.selenium.safari.SafariDriver;
34  
35  /** Enumerates the available web drivers for Selenium 2.
36   * @author Volker Bergmann */
37  public class Selenium2Driver {
38  
39      // Note: Needed to avoid 'enum' approach since this would make plexus' generate-metadata goal fail which considers a method
40      // call in an enum constructor call to be a syntax error
41  
42      private static final Map<String, Selenium2Driver> INSTANCES = new HashMap<String, Selenium2Driver>();
43  
44      // Constants for the instances ---------------------------------------------
45  
46      /** Firefox driver */
47      public static Selenium2Driver FIREFOX = new Selenium2Driver("FIREFOX", BrowserType.FIREFOX, FirefoxDriver.class,
48              DesiredCapabilities.firefox());
49  
50      /** Internet Explorer driver */
51      public static Selenium2Driver INTERNET_EXPLORER = new Selenium2Driver("INTERNET_EXPLORER", BrowserType.IE, InternetExplorerDriver.class,
52              DesiredCapabilities.internetExplorer());
53  
54      /** HTMLUnit driver */
55      public static Selenium2Driver HTML_UNIT = new Selenium2Driver("HTML_UNIT", BrowserType.HTMLUNIT, HtmlUnitDriver.class,
56              DesiredCapabilities.htmlUnit());
57  
58      /** Google Chrome driver */
59      public static Selenium2Driver CHROME = new Selenium2Driver("CHROME", BrowserType.GOOGLECHROME, ChromeDriver.class, createChromeCaps());
60  
61      /** Safari driver */
62      public static Selenium2Driver SAFARI = new Selenium2Driver("SAFARI", BrowserType.SAFARI, SafariDriver.class, DesiredCapabilities.safari());
63  
64      /** PhantomJS driver */
65      public static Selenium2Driver PHANTOMJS = new Selenium2Driver("PHANTOMJS", BrowserType.PHANTOMJS, RemoteWebDriver.class,
66              DesiredCapabilities.phantomjs());
67  
68      // attributes --------------------------------------------------------------
69  
70      private final String browserName;
71      private final Class<? extends WebDriver> driverClass;
72      private final DesiredCapabilities capabilities;
73  
74      // constructor -------------------------------------------------------------
75  
76      private Selenium2Driver(String driverName, String browserName, Class<? extends WebDriver> driverClass,
77              DesiredCapabilities capabilities) {
78          this.browserName = browserName;
79          this.driverClass = driverClass;
80          this.capabilities = capabilities;
81          INSTANCES.put(driverName, this);
82      }
83  
84      // public interface --------------------------------------------------------
85  
86      /** @return the name of the related browser */
87      public String getBrowserName() {
88          return browserName;
89      }
90  
91      /** @return a freshly created instance of the related WebDriver class */
92      public WebDriver newLocalDriver() {
93          return BeanUtil.newInstance(driverClass);
94      }
95  
96      /** @param url the URL for which to create a WebDriver instance.
97       * @param arguments Additional arguments to include for the WebDriver instance, or an empty array.
98       * @return a freshly created instance of the related WebDriver class */
99      public WebDriver newRemoteDriver(URL url, String[] arguments) {
100         AludraSeleniumHttpCommandExecutor executor = new AludraSeleniumHttpCommandExecutor(url);
101 
102         DesiredCapabilities caps = capabilities;
103         if (arguments != null && arguments.length > 0) {
104             caps = new DesiredCapabilities(capabilities);
105             ChromeOptions opts = (ChromeOptions) caps.getCapability(ChromeOptions.CAPABILITY);
106             if (opts != null) {
107                 opts.addArguments(arguments);
108             }
109         }
110 
111         RemoteWebDriver driver = new RemoteWebDriver(executor, caps);
112         driver.setFileDetector(new LocalFileDetector());
113         return driver;
114     }
115 
116     // public static interface -------------------------------------------------
117 
118     /** @param driverName the name of the driver
119      * @return the driver instance associated with the driverName
120      * @throws IllegalArgumentException if no driver is configured with this name */
121     public static Selenium2Driver valueOf(String driverName) {
122         Selenium2Driver driver = INSTANCES.get(driverName);
123         if (driver == null) {
124             throw new IllegalArgumentException("Unknown driver: " + driverName);
125         }
126         return driver;
127     }
128 
129     // private helper methods --------------------------------------------------
130 
131     private static DesiredCapabilities createChromeCaps() {
132         DesiredCapabilities caps = DesiredCapabilities.chrome();
133         ChromeOptions opts = new ChromeOptions();
134         opts.addArguments("--disable-extensions");
135         caps.setCapability(ChromeOptions.CAPABILITY, opts);
136         return caps;
137     }
138 
139 }
