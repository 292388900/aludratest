16  package org.aludratest.service.cmdline;
17  
18  import java.io.ByteArrayOutputStream;
19  import java.io.OutputStream;
20  
21  import org.aludratest.util.data.ByteArrayData;
22  import org.aludratest.util.data.StringData;
23  import org.databene.commons.Validator;
24  
25  /** Common parent class for the output streams of a process: stdout and errout.
26   * @author Volker Bergmann */
27  public abstract class Out {
28  
29      final CommandLineProcess<?> process;
30  
31      // construction ------------------------------------------------------------
32  
33      protected Out(CommandLineProcess<?> process) {
34          this.process = process;
35      }
36  
37      // operational interface ---------------------------------------------------
38  
39      /** Redirects the process' output to a stream.
40       * @param textOutput
41       * @return this */
42      public Out redirectTo(StringData textOutput) {
43          ByteArrayData buffer = new ByteArrayData();
44          redirectTo(buffer);
45          textOutput.setValue(new String(buffer.getValue()));
46          return this;
47      }
48  
49      /** Redirects the process' stdout to a stream.
50       * @param byteOutput
51       * @return this */
52      public Out redirectTo(ByteArrayData byteOutput) {
53          ByteArrayOutputStream buffer = new ByteArrayOutputStream();
54          redirectTo(buffer);
55          byteOutput.setValue(buffer.toByteArray());
56          return this;
57      }
58  
59      /** Redirects the output to a stream.
60       * @param out the target stream
61       * @return this */
62      public abstract Out redirectTo(OutputStream out);
63  
64      /** @return the next line available in this stream */
65      public Line nextLine() {
66          return new Line(this);
67      }
68  
69      /** Asserts that the validator matches the next line of the output.
70       * @param validator the matcher
71       * @return this */
72      public abstract Out assertNextLineMatches(Validator<String> validator);
73  
74      /** Skips the output lines until a line matches the validator or the end of the file is reached.
75       * @param validator the matcher
76       * @return a Line object for the first matching line */
77      public abstract Line skipUntilLineMatches(Validator<String> validator);
78  
79      /** Asserts that the output stream is empty
80       * @return this */
81      public abstract Out assertEmpty();
82  
83  }
