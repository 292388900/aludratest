16  package org.aludratest.scheduler;
17  
18  import org.aludratest.scheduler.node.RunnerGroup;
19  import org.aludratest.scheduler.node.RunnerLeaf;
20  import org.aludratest.testcase.event.TestStepInfo;
21  
22  /** Interface for listeners wanting to be notified about the progress of a testing run. Please always keep in mind that test cases
23   * could be executed in parallel, so the listeners should not cause any locks or keep an object state. <br>
24   * <br>
25   * To register a RunnerListener, retrieve the <code>RunnerListenerRegistry</code> e.g. from the AludraTest ServiceManager
26   * instance, and add it to this registry:
27   * 
28   * <pre>
29   * AludraTest myAludra = AludraTest.startFramework(MyTestSuite.class);
30   * RunnerListenerRegistry registry = myAludra.getServiceManager().newImplementorInstance(RunnerListenerRegistry.class);
31   * registry.addRunnerListener(myRunnerListener);
32   * </pre>
33   * 
34   * @author falbrech */
35  public interface RunnerListener {
36  
37      /** Called when the overall test process is about to be started, i.e. the first test case is about to run.
38       * 
39       * @param runnerTree RunnerTree which contains all test cases for execution. */
40      void startingTestProcess(RunnerTree runnerTree);
41  
42      /** Called when the given test group is about to be started, i.e. the first test case within the group is about to run.
43       * 
44       * @param runnerGroup RunnerGroup which contains test cases for execution. */
45      void startingTestGroup(RunnerGroup runnerGroup);
46  
47      /** Called when the given test case is about to be run.
48       * 
49       * @param runnerLeaf RunnerLeaf which encapsulates the single test case which is about to be run. */
50      void startingTestLeaf(RunnerLeaf runnerLeaf);
51  
52      /** Calles when the given test case has finished execution. This is also called if some or all steps of the test case failed,
53       * or the test case is ignored.
54       * 
55       * @param runnerLeaf RunnerLeaf which encapsulates the single test case which has finished execution. */
56      void finishedTestLeaf(RunnerLeaf runnerLeaf);
57  
58      /** Called when all test cases of the given runner group have finished execution.
59       * 
60       * @param runnerGroup RunnerGroup of which all test cases have finished execution. */
61      void finishedTestGroup(RunnerGroup runnerGroup);
62  
63      /** Called when all test cases have finished execution.
64       * 
65       * @param runnerTree RunnerTree which contains all test cases. */
66      void finishedTestProcess(RunnerTree runnerTree);
67  
68      /** Called when a new test step group within the given test case node starts. Note that there is no matching
69       * "endTestStepGroup"; every following test step is intended to be assigned to the currently open test step group for the
70       * given test case.
71       * 
72       * @param runnerLeaf RunnerLeaf encapsulating the test step for which a new test step group shall be opened.
73       * @param groupName Name of the group to start. */
74      void newTestStepGroup(RunnerLeaf runnerLeaf, String groupName);
75  
76      /** Called when a single test step within a test case has been performed. This is the smallest traceable unit within
77       * AludraTest. The step should be assigned to the last test step group started for the given RunnerLeaf. If there is no test
78       * step group active, it is OK to throw an IllegalStateException.
79       * 
80       * @param runnerLeaf RunnerLeaf to which to assign the given test step.
81       * @param testStepInfo Information about the test step. */
82      void newTestStep(RunnerLeaf runnerLeaf, TestStepInfo testStepInfo);
83  
84  }
