16  package org.aludratest.scheduler.impl;
17  
18  import java.text.ParseException;
19  import java.util.ArrayList;
20  import java.util.Collections;
21  import java.util.List;
22  import java.util.regex.Matcher;
23  import java.util.regex.Pattern;
24  
25  import org.aludratest.scheduler.TestClassFilter;
26  
27  /** Parses a filter string.
28   * 
29   * @author falbrech */
30  public class FilterParser {
31  
32      /** Parses the given filter string.
33       * 
34       * @param inputString Filter string to parse.
35       * @return A test class filter representing the filter string.
36       * 
37       * @throws ParseException If a parsing error occurs, i.e. the input filter string is invalid. */
38      public TestClassFilter parse(String inputString) throws ParseException {
39          List<Token> tokens = new FilterTokenizer().tokenize(inputString);
40          return parseOrFilter(tokens);
41      }
42  
43      private TestClassFilter parseOrFilter(List<Token> tokens) throws ParseException {
44          List<TestClassFilter> filters = new ArrayList<TestClassFilter>();
45  
46          while (!tokens.isEmpty()) {
47              filters.add(parseAndFilter(tokens));
48              if (!tokens.isEmpty()) {
49                  consumeToken(tokens, FilterTokenizer.TOKEN_OR);
50                  assertNotEmpty(tokens);
51              }
52          }
53  
54          return filters.size() == 1 ? filters.get(0) : new OrTestClassFilter(filters);
55      }
56  
57      private TestClassFilter parseAndFilter(List<Token> tokens) throws ParseException {
58          assertNotEmpty(tokens);
59  
60          List<TestClassFilter> filters = new ArrayList<TestClassFilter>();
61  
62          while (!tokens.isEmpty() && tokens.get(0).token != FilterTokenizer.TOKEN_OR) {
63              // get attribute name
64              Token token = consumeToken(tokens, FilterTokenizer.TOKEN_IDENTIFIER);
65              String attributeName = token.sequence.trim();
66              token = consumeToken(tokens, 0);
67              boolean invert = false;
68              if (token.token == FilterTokenizer.TOKEN_NOT_EQUAL) {
69                  invert = true;
70              }
71              else {
72                  assertToken(token, FilterTokenizer.TOKEN_EQUAL);
73              }
74              List<String> values = parseValues(tokens);
75              filters.add(new AttributeBasedTestClassFilter(attributeName, values, invert));
76              if (!tokens.isEmpty() && tokens.get(0).token != FilterTokenizer.TOKEN_OR) {
77                  consumeToken(tokens, FilterTokenizer.TOKEN_SEMICOLON);
78              }
79          }
80  
81          if (filters.isEmpty() && !tokens.isEmpty()) {
82              throw new ParseException("Unexpected token " + tokens.get(0) + " at offset " + tokens.get(0).offset,
83                      tokens.get(0).offset);
84          }
85  
86          return new AndTestClassFilter(filters);
87      }
88  
89      private List<String> parseValues(List<Token> tokens) throws ParseException {
90          assertNotEmpty(tokens);
91  
92          // either bracket, or single value
93          Token token = consumeToken(tokens, 0);
94          if (token.token == FilterTokenizer.TOKEN_OPEN_BRACKET) {
95              List<String> values = parseValueList(tokens);
96              consumeToken(tokens, FilterTokenizer.TOKEN_CLOSE_BRACKET);
97              return values;
98          }
99  
100         assertToken(token, FilterTokenizer.TOKEN_IDENTIFIER);
101         return Collections.singletonList(token.sequence.trim());
102     }
103 
104     private List<String> parseValueList(List<Token> tokens) throws ParseException {
105         assertNotEmpty(tokens);
106 
107         List<String> result = new ArrayList<String>();
108 
109         while (tokens.get(0).token != FilterTokenizer.TOKEN_CLOSE_BRACKET) {
110             Token token = consumeToken(tokens, FilterTokenizer.TOKEN_IDENTIFIER);
111             result.add(token.sequence.trim());
112             assertNotEmpty(tokens);
113             if (tokens.get(0).token != FilterTokenizer.TOKEN_CLOSE_BRACKET) {
114                 consumeToken(tokens, FilterTokenizer.TOKEN_COMMA);
115             }
116         }
117 
118         return result;
119     }
120 
121     private Token consumeToken(List<Token> tokens, int expectedType) throws ParseException {
122         assertNotEmpty(tokens);
123 
124         Token token = tokens.remove(0);
125         if (expectedType > 0) {
126             assertToken(token, expectedType);
127         }
128         return token;
129     }
130 
131     private void assertToken(Token token, int expectedType) throws ParseException {
132         if (token.token != expectedType) {
133             throw new ParseException("Unexpected token " + token.sequence + " at offset " + token.offset, token.offset);
134         }
135     }
136 
137     private void assertNotEmpty(List<Token> tokens) throws ParseException {
138         if (tokens.isEmpty()) {
139             throw new ParseException("Unexpected end of filter string", 0);
140         }
141     }
142 
143     private static class FilterTokenizer {
144 
145         private static final int TOKEN_OR = 1;
146 
147         private static final int TOKEN_SEMICOLON = 2;
148 
149         private static final int TOKEN_OPEN_BRACKET = 3;
150 
151         private static final int TOKEN_CLOSE_BRACKET = 4;
152 
153         private static final int TOKEN_IDENTIFIER = 5;
154 
155         private static final int TOKEN_EQUAL = 6;
156 
157         private static final int TOKEN_NOT_EQUAL = 7;
158 
159         private static final int TOKEN_COMMA = 8;
160 
161         private List<TokenInfo> tokenInfos = new ArrayList<TokenInfo>();
162 
163         public FilterTokenizer() {
164             addTokenInfo("\\|", TOKEN_OR);
165             addTokenInfo(";", TOKEN_SEMICOLON);
166             addTokenInfo(",", TOKEN_COMMA);
167             addTokenInfo("\\(", TOKEN_OPEN_BRACKET);
168             addTokenInfo("\\)", TOKEN_CLOSE_BRACKET);
169             addTokenInfo("!=", TOKEN_NOT_EQUAL);
170             addTokenInfo("=", TOKEN_EQUAL);
171             addTokenInfo("([^\\|;=!\\(\\)\\[\\],]+)|\\[\\]", TOKEN_IDENTIFIER);
172         }
173 
174         public List<Token> tokenize(String input) throws ParseException {
175             List<Token> result = new ArrayList<Token>();
176 
177             int offset = 0;
178             while (!"".equals(input)) {
179                 // remove spaces before and after each token
180                 while (input.startsWith(" ")) {
181                     input = input.substring(1);
182                     offset++;
183                 }
184                 if ("".equals(input)) {
185                     break;
186                 }
187 
188                 boolean match = false;
189                 for (TokenInfo info : tokenInfos) {
190                     Matcher m = info.regex.matcher(input);
191                     if (m.find()) {
192                         match = true;
193                         String tok = m.group();
194                         result.add(new Token(info.token, tok, offset));
195                         input = m.replaceFirst("");
196                         offset += tok.length();
197                         break;
198                     }
199                 }
200 
201                 if (!match) {
202                     throw new ParseException("Unexpected character '" + input.charAt(0) + "' at offset " + offset, offset);
203                 }
204             }
205 
206             return result;
207         }
208 
209         private void addTokenInfo(String regex, int token) {
210             tokenInfos.add(new TokenInfo(Pattern.compile("^(" + regex + ")"), token));
211         }
212     }
213 
214     private static class TokenInfo {
215 
216         private final Pattern regex;
217 
218         private final int token;
219 
220         public TokenInfo(Pattern regex, int token) {
221             this.regex = regex;
222             this.token = token;
223         }
224     }
225 
226     private static class Token {
227         public final int token;
228         public final String sequence;
229         public final int offset;
230 
231         public Token(int token, String sequence, int offset) {
232             this.token = token;
233             this.sequence = sequence;
234             this.offset = offset;
235         }
236     }
237 }
