63  /** An XML based Test Data provider. XML files must have the AludraTest XML Testdata format; best use AludraTest VDE Plugin for
64   * eclipse to create testdata files. The uri value of Source annotations must point to an XML file. This file is searched in these
65   * two locations:
66   * <ol>
67   * <li>In a folder <i>xlsRootPath</i>/package/of/testclass/as/folder/MyTestClass/</li>
68   * <li>Relative to <i>xlsRootPath</i>
69   * </ol>
70   * xlsRootPath value is configured in aludratest.properties.
71   * 
72   * @author falbrech */
73  public class XmlBasedTestDataProvider implements TestDataProvider {

81      @Override
82      public List<TestCaseData> getTestDataSets(Method method) {
83          if (method.getParameterTypes().length == 0) {
84              return Collections.singletonList(new TestCaseData("0", null, null));
85          }
86  
87          // get @Source annotated parameters
88          Annotation[][] annots = method.getParameterAnnotations();
89  
90          Offset offsetAnno = method.getAnnotation(Offset.class);
91          boolean ignored = method.getAnnotation(Ignored.class) != null;
92          String ignoredReason = null;
93          if (ignored) {
94              ignoredReason = method.getAnnotation(Ignored.class).value();
95              if ("".equals(ignoredReason)) {
96                  ignoredReason = null;
97              }
98          }
99  
100         List<TestCaseData> result = new ArrayList<TestCaseData>();
101 
102         // cache for loaded file models
103         Map<String, TestData> loadedFileModels = new HashMap<String, TestData>();
104 
105         // load param by param; transpose into test case data afterwards
106         List<List<Data>> allData = new ArrayList<List<Data>>();
107         for (int i = 0; i < annots.length; i++) {
108             List<Data> paramData = getDataObjects(method, i, loadedFileModels);
109             if (offsetAnno != null) {
110                 int offset = offsetAnno.value();
111                 if (offset < paramData.size()) {
112                     paramData = paramData.subList(offset, paramData.size());
113                 }
114                 else {
115                     paramData = Collections.emptyList();
116                 }
117             }
118             allData.add(paramData);
119         }
120 
121         // transposition with the help of first loaded XML file
122         Source firstSource = getRequiredSourceAnnotation(annots[0], "first parameter");
123         TestData data = loadedFileModels.get(firstSource.uri());
124 
125         List<TestDataConfiguration> configs = data.getConfigurations();
126         for (int i = 0; i < configs.size(); i++) {
127             TestDataConfiguration config = configs.get(i);
128 
129             List<Data> dataForConfig = new ArrayList<Data>();
130             // ensure that all data lists contain enough entries
131             for (List<Data> ls : allData) {
132                 if (ls.size() <= i) {
133                     result.add(new TestCaseData(getNextAutoId(result, false), new AutomationException("For method " + method
134                             + ", not all referenced XML files contain the same amount of test configurations.")));
135                     dataForConfig = null;
136                     break;
137                 }
138                 else {
139                     dataForConfig.add(ls.get(i));
140                 }
141             }
142 
143             if (dataForConfig != null) {
144                 if (ignored) {
145                     result.add(new TestCaseData(config.getName(), null, dataForConfig.toArray(new Data[0]), true, ignoredReason));
146                 }
147                 else {
148                     result.add(new TestCaseData(config.getName(), null, dataForConfig.toArray(new Data[0]), config.isIgnored(),
149                             config.isIgnored() ? config.getIgnoredReason() : null));
150                 }
151             }
152         }
153 
154         return result;
155     }

157     private List<Data> getDataObjects(Method method, int paramIndex, Map<String, TestData> loadedFileModels) {
158         Annotation[][] annots = method.getParameterAnnotations();
159         String paramName = method.getName() + " param #" + paramIndex;
160         Source src = getRequiredSourceAnnotation(annots[paramIndex], paramName);
161 
162         // try to load the referenced file, or reuse from cache
163         String uri = src.uri();
164         if (uri == null || "".equals(uri)) {
165             throw new AutomationException("@Source annotation does not specify required uri parameter");
166         }
167         if (src.segment() == null || "".equals(src.segment())) {
168             throw new AutomationException("@Source annotation does not specify required segment parameter");
169         }
170 
171         TestData testData;
172         if (loadedFileModels.containsKey(uri)) {
173             testData = loadedFileModels.get(uri);
174         }
175         else {
176             InputStream in = null;
177             try {
178                 // real URI?
179                 in = tryFindXml(uri, method);
180                 testData = TestData.read(in);
181                 // some base validations
182                 if (testData.getMetadata() == null || testData.getMetadata().getSegments() == null
183                         || testData.getConfigurations() == null) {
184                     throw new AutomationException("Test data XML " + uri + " has an invalid format or is incomplete.");
185                 }
186                 loadedFileModels.put(uri, testData);
187             }
188             catch (Exception e) {
189                 throw new AutomationException("Could not read test data XML at " + uri, e);
190             }
191             finally {
192                 IOUtils.closeQuietly(in);
193             }
194         }
195 
196         List<Data> dataElements = new ArrayList<Data>();
197 
198         // get metadata for requested segment
199         TestDataSegmentMetadata segmentMeta = null;
200         for (TestDataSegmentMetadata segment : testData.getMetadata().getSegments()) {
201             if (segment.getName().equals(src.segment())) {
202                 segmentMeta = segment;
203                 break;
204             }
205         }
206         if (segmentMeta == null) {
207             throw new AutomationException("Could not find segment " + src.segment() + " in XML file " + uri);
208         }
209 
210         // for each configuration entry, find values
211         for (TestDataConfiguration config : testData.getConfigurations()) {
212             if (containsSegment(config, segmentMeta.getName())) {
213                 dataElements.add(buildObject(testData, config, segmentMeta.getName()));
214             }
215             else {
216                 dataElements.add(null);
217             }
218         }
219 
220         return dataElements;
221     }

223     private TestDataSegmentMetadata getSegmentMetadata(TestData testData, String segmentName) {
224         for (TestDataSegmentMetadata segment : testData.getMetadata().getSegments()) {
225             if (segmentName.equals(segment.getName())) {
226                 return segment;
227             }
228         }
229         return null;
230     }

232     private Object getFieldValue(TestDataConfiguration configuration, String segmentName, TestDataFieldMetadata fieldMeta) {
233         String fieldName = fieldMeta.getName();
234         for (TestDataConfigurationSegment segment : configuration.getSegments()) {
235             if (segmentName.equals(segment.getName())) {
236                 for (TestDataFieldValue field : segment.getFieldValues()) {
237                     if (fieldName.equals(field.getName())) {
238                         Object value = field.getFieldValueAsJavaType();
239                         if (field.isScript() && (value instanceof String)) {
240                             return new ScriptToEvaluate(value.toString(), fieldMeta.getFormatterPattern(),
241                                     toLocale(fieldMeta.getFormatterLocale()));
242                         }
243                         return value;
244                     }
245                 }
246             }
247         }
248         return null;
249     }

251     private boolean containsSegment(TestDataConfiguration configuration, String segmentName) {
252         for (TestDataConfigurationSegment segment : configuration.getSegments()) {
253             if (segmentName.equals(segment.getName())) {
254                 return true;
255             }
256         }
257         return false;
258     }

260     private Data buildObject(TestData testData, TestDataConfiguration configuration, String segmentName) {
261         TestDataSegmentMetadata segmentMeta = getSegmentMetadata(testData, segmentName);
262         if (segmentMeta == null) {
263             throw new AutomationException("Segment " + segmentName + " is not defined in test data XML metadata.");
264         }
265 
266         if (!containsSegment(configuration, segmentName)) {
267             return null;
268         }
269 
270         String dcn = segmentMeta.getDataClassName();
271 
272         try {
273             @SuppressWarnings("unchecked")
274             Class<? extends Data> clazz = (Class<? extends Data>) Class.forName(dcn);
275             Data data = clazz.newInstance();
276 
277             // populate data
278             Map<String, ScriptToEvaluate> scriptValues = new HashMap<String, ScriptToEvaluate>();
279             Map<String, Object> plainValues = new HashMap<String, Object>();
280 
281             for (TestDataFieldMetadata field : segmentMeta.getFields()) {
282                 Object value;
283                 // if field is reference to another segment or segments, recurse into object creation
284                 if (field.getType() == TestDataFieldType.OBJECT) {
285                     value = buildObject(testData, configuration, segmentMeta.getName() + "." + field.getName());
286                 }
287                 else if (field.getType() == TestDataFieldType.OBJECT_LIST) {
288                     value = buildObjectList(testData, configuration, segmentMeta.getName() + "." + field.getName());
289                 }
290                 else {
291                     value = getFieldValue(configuration, segmentName, field);
292                 }
293 
294                 if (!(value instanceof ScriptToEvaluate)) {
295                     // also put null in map to avoid "no such reference"
296                     plainValues.put(field.getName(), value);
297                 }
298                 else {
299                     scriptValues.put(field.getName(), (ScriptToEvaluate) value);
300                 }
301 
302                 if (value != null && !(value instanceof ScriptToEvaluate)) {
303                     BeanUtil.setPropertyValue(data, field.getName(), value);
304                 }
305             }
306 
307             // now evaluate scripts. Offer already calculated fields as Context variables
308             int lastErrorCount;
309             int errorCount = 0;
310             AutomationException lastException = null;
311             do {
312                 lastErrorCount = errorCount;
313                 errorCount = 0;
314                 Iterator<Map.Entry<String, ScriptToEvaluate>> scriptIter = scriptValues.entrySet().iterator();
315                 while (scriptIter.hasNext()) {
316                     Map.Entry<String, ScriptToEvaluate> entry = scriptIter.next();
317                     try {
318                         Object value = evaluate(entry.getValue().script, entry.getValue().formatPattern,
319                                 entry.getValue().formatLocale, plainValues);
320                         plainValues.put(entry.getKey(), value);
321                         if (value != null) {
322                             BeanUtil.setPropertyValue(data, entry.getKey(), value);
323                         }
324                         scriptIter.remove();
325                     }
326                     catch (AutomationException e) {
327                         lastException = e;
328                         errorCount++;
329                     }
330                 }
331             }
332             while (errorCount != lastErrorCount && !scriptValues.isEmpty());
333 
334             if (errorCount > 0 && lastException != null) {
335                 throw lastException;
336             }
337 
338             return data;
339         }
340         catch (Exception e) {
341             throw new AutomationException("Could create data object for segment " + segmentMeta.getName(), e);
342         }
343     }

345     private List<Data> buildObjectList(TestData testData, TestDataConfiguration configuration, String segmentName) {
346         List<Data> result = new ArrayList<Data>();
347 
348         for (int i = 1; i < 1000; i++) {
349             String segName = segmentName + "-" + i;
350             if (!containsSegment(configuration, segName)) {
351                 break;
352             }
353 
354             result.add(buildObject(testData, configuration, segName));
355         }
356 
357         return result;
358     }

360     private Source getRequiredSourceAnnotation(Annotation[] annotations, String parameterName) {
361         for (Annotation anno : annotations) {
362             if (anno instanceof Source) {
363                 return (Source) anno;
364             }
365         }
366         throw new AutomationException("Parameter does not have a @Source annotation: " + parameterName);
367     }

369     private InputStream tryFindXml(String uri, Method testMethod) throws IOException, AutomationException, URISyntaxException {
370         if (uri.matches("[a-z]+://.*")) {
371             URI realUri = new URI(uri);
372             URL url = realUri.toURL();
373             return url.openStream();
374         }
375 
376         // first search try: full path, as used for Excels
377         StringBuilder sbPath = new StringBuilder();
378         sbPath.append(aludraConfig.getXlsRootPath());
379 
380         if (sbPath.toString().endsWith(File.separator)) {
381             sbPath.delete(sbPath.length() - 1, sbPath.length());
382         }
383         sbPath.append(File.separator);
384         sbPath.append(testMethod.getDeclaringClass().getName().replace(".", File.separator));
385         sbPath.append(File.separator);
386         sbPath.append(uri);
387         File f = new File(sbPath.toString());
388         if (f.isFile()) {
389             return new FileInputStream(f);
390         }
391 
392         // second try: directly under root path
393         sbPath = new StringBuilder();
394         sbPath.append(aludraConfig.getXlsRootPath());
395 
396         if (sbPath.toString().endsWith(File.separator)) {
397             sbPath.delete(sbPath.length() - 1, sbPath.length());
398         }
399         sbPath.append(File.separator);
400         sbPath.append(uri);
401         f = new File(sbPath.toString());
402         if (f.isFile()) {
403             return new FileInputStream(f);
404         }
405 
406         throw new AutomationException("Could not find test data XML file " + uri);
407     }

409     private String getNextAutoId(List<TestCaseData> dataSets, boolean error) {
410         String prefix = error ? "error-" : "";
411         int nextAutoId = dataSets.size();
412         boolean found;
413         do {
414             found = false;
415             for (TestCaseData tcd : dataSets) {
416                 if (tcd.getId().equals(prefix + nextAutoId)) {
417                     found = true;
418                     break;
419                 }
420             }
421             if (found) {
422                 nextAutoId++;
423             }
424         }
425         while (found);
426         return prefix + nextAutoId;
427     }

429     /** Evaluates the given data script, applying the given format pattern and locale, if specified.
430      * 
431      * @param script Script to evaluate, e.g. <code>addDaysToNow(5)</code>
432      * @param formatPattern Format pattern to apply. Can be a format accepted by <code>SimpleDateFormat</code> or
433      *            <code>DecimalFormat</code>, depending on type of expression. If not specified, a type-specific default format is
434      *            used. If the expression evaluates to a String, this parameter is ignored.
435      * @param locale Locale to apply to the format pattern. If not specified, <code>Locale.US</code> is used (<b>NOT</b> the
436      *            platform default, to ensure platform-independent operation).
437      * @param contextVariables Map with objects which should be offered in the script context as variables. Can be
438      *            <code>null</code>.
439      * @return */
440     public String evaluate(String script, String formatPattern, Locale locale, Map<String, Object> contextVariables) {
441         if (locale == null) {
442             locale = Locale.US;
443         }
444         Context context = Context.enter();
445 
446         try {
447             Scriptable scope = context.initStandardObjects();
448 
449             // find all script libraries and add their scripts
450             for (ScriptLibrary lib : scriptLibraries.values()) {
451                 lib.addFunctionsToContext(context, scope);
452             }
453 
454             // put context variables
455             if (contextVariables != null) {
456                 for (Map.Entry<String, Object> entry : contextVariables.entrySet()) {
457                     scope.put(entry.getKey(), scope, entry.getValue() == null ? null : Context.toObject(entry.getValue(), scope));
458                 }
459             }
460 
461             try {
462                 Object result = context.evaluateString(scope, script, "<cmd>", 1, null);
463                 if (result instanceof Undefined) {
464                     return null;
465                 }
466                 result = toJavaObject(result);
467 
468                 // apply patterns, if required
469                 if (result instanceof Date) {
470                     if (formatPattern == null) {
471                         formatPattern = "yyyy-MM-dd";
472                     }
473                     SimpleDateFormat sdf = new SimpleDateFormat(formatPattern, locale);
474                     return sdf.format(result);
475                 }
476                 if (result instanceof Number) {
477                     if (formatPattern == null) {
478                         formatPattern = "#.#";
479                     }
480                     DecimalFormat df = new DecimalFormat(formatPattern, DecimalFormatSymbols.getInstance(locale));
481                     return df.format(result);
482                 }
483 
484                 return result.toString();
485             }
486             catch (RhinoException e) {
487                 throw new AutomationException("Cannot evaluate test data script '" + script + "'", e);
488             }
489         }
490         finally {
491             Context.exit();
492         }
493     }

495     private Object toJavaObject(Object jsObject) {
496         if (!(jsObject instanceof ScriptableObject)) {
497             // already Java
498             return jsObject;
499         }
500         // analyze object class name to determine target type
501         if (jsObject.getClass().getName().endsWith("Date")) {
502             return Context.jsToJava(jsObject, Date.class);
503         }
504         if (jsObject.getClass().getName().endsWith("Number")) {
505             return Context.jsToJava(jsObject, Double.class);
506         }
507         return Context.toString(jsObject);
508     }

510     private static Locale toLocale(String s) {
511         if (s == null || "".equals(s)) {
512             return null;
513         }
514 
515         // language only?
516         if (s.matches("[a-z]{2}")) {
517             return new Locale(s);
518         }
519 
520         // language and country?
521         if (s.matches("[a-z]{2}_[A-Z]{2}")) {
522             String[] parts = s.split("_");
523             return new Locale(parts[0], parts[1]);
524         }
525 
526         // variant?
527         if (s.matches("[a-z]{2}_[A-Z]{2}_[^_]+")) {
528             String[] parts = s.split("_");
529             return new Locale(parts[0], parts[1], parts[2]);
530         }
531 
532         // invalid locale string
533         throw new IllegalArgumentException("Invalid Locale value found in XML: " + s);
534     }
