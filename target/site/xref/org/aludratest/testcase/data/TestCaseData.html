16  package org.aludratest.testcase.data;
17  
18  import org.aludratest.dict.Data;
19  import org.databene.commons.ArrayUtil;
20  
21  /** Represents the data for a single invocation of a method of a test class. Conceptually spoken, a concrete instance of this class
22   * <b>is</b> the "test case". More formally, the combination of a test method <i>M</i> and a <code>TestCaseData</code> object
23   * <i>D</i> makes up the test case <i>T</i>: <code>T = (M, D)</code> or <code>T = M(D)</code>.
24   * 
25   * @author falbrech */
26  public final class TestCaseData {
27  
28      private String id;
29  
30      private String description;
31  
32      private Data[] data;
33  
34      private boolean ignored;
35  
36      private String ignoredReason;
37  
38      private Throwable exception;
39  
40      /** Constructs a new TestCaseData object.
41       * 
42       * @param id ID for the test case data. Must be not <code>null</code> and unique within all datasets for the same test method
43       *            (this will be checked by the framework before test invocation).
44       * @param description Description for the test case data. May be <code>null</code>.
45       * @param data The array of parameters for the test method to use for a single test invocation. May be <code>null</code> if
46       *            the method does not require any parameters. Otherwise, it must have as many entries as the method has
47       *            parameters. */
48      public TestCaseData(String id, String description, Data[] data) {
49          this(id, description, data, false);
50      }
51  
52      /** Constructs a new TestCaseData object.
53       * 
54       * @param id ID for the test case data. Must be not <code>null</code> and unique within all datasets for the same test method
55       *            (this will be checked by the framework before test invocation).
56       * @param description Description for the test case data. May be <code>null</code>.
57       * @param data The array of parameters for the test method to use for a single test invocation. May be <code>null</code> if
58       *            the method does not require any parameters. Otherwise, it must have as many entries as the method has
59       *            parameters.
60       * @param ignored If <code>true</code>, marks this data set as ignored, i.e. this test case shall not be invoked (this
61       *            behaviour can be disabled by Framework settings). */
62      public TestCaseData(String id, String description, Data[] data, boolean ignored) {
63          this.id = id;
64          this.description = description;
65          this.data = data == null ? new Data[0] : ArrayUtil.copyOfRange(data, 0, data.length);
66          this.ignored = ignored;
67      }
68  
69      /** Constructs a new TestCaseData object.
70       * 
71       * @param id ID for the test case data. Must be not <code>null</code> and unique within all datasets for the same test method
72       *            (this will be checked by the framework before test invocation).
73       * @param description Description for the test case data. May be <code>null</code>.
74       * @param data The array of parameters for the test method to use for a single test invocation. May be <code>null</code> if
75       *            the method does not require any parameters. Otherwise, it must have as many entries as the method has
76       *            parameters.
77       * @param ignored If <code>true</code>, marks this data set as ignored, i.e. this test case shall not be invoked (this
78       *            behaviour can be disabled by Framework settings).
79       * @param ignoredReason A text describing why this data set has been marked as ignored. */
80      public TestCaseData(String id, String description, Data[] data, boolean ignored, String ignoredReason) {
81          this(id, description, data, ignored);
82          this.ignoredReason = ignoredReason;
83      }
84  
85      /** Constructs a new TestCaseData object, for a dataset which could not be loaded for a given reason.
86       * 
87       * @param id ID for the test case data. Must be not <code>null</code> and unique within all datasets for the same test method
88       *            (this will be checked by the framework before test invocation).
89       * @param exception The cause why the data for this dataset could not be loaded. */
90      public TestCaseData(String id, Throwable exception) {
91          this.id = id;
92          this.exception = exception;
93      }
94  
95      /** Returns a unique ID for this test case dataset (unique within the parent test suite, represented by the test class). Can
96       * also contain a short description, e.g. <code>A3456_max_length_for_street</code>. Should not contain space characters!
97       * 
98       * @return A unique ID for this test case dataset, never <code>null</code>. */
99      public String getId() {
100         return id;
101     }
102 
103     /** Returns a textual description for this test case dataset. May be <code>null</code> if no description is available.
104      * 
105      * @return A textual description for this test case dataset, or <code>null</code>. */
106     public String getDescription() {
107         return description;
108     }
109 
110     /** Returns the data elements for this test case, in the order of the parameter list of the test case method. May be an empty
111      * array if the test case method does not require any parameters. Otherwise, it must have as many entries as the method's
112      * parameter list.
113      * 
114      * @return The data elements for this test case, maybe empty, but never <code>null</code>. */
115     public Data[] getData() {
116         return data;
117     }
118 
119     /** Returns <code>true</code> if this dataset should not be tested, but skipped.
120      * 
121      * @return <code>true</code> if this dataset should not be tested, but skipped, <code>false</code> otherwise. */
122     public boolean isIgnored() {
123         return ignored;
124     }
125 
126     /** Returns the optional reason why this dataset has been marked as ignored.
127      * 
128      * @return The optional reason why this dataset has been marked as ignored, or <code>null</code>. */
129     public String getIgnoredReason() {
130         return ignoredReason;
131     }
132 
133     /** Returns an exception that describes why this dataset could not be loaded. If <code>null</code> is returned, the dataset
134      * could be loaded successfully.
135      * 
136      * @return <code>null</code>, or an exception that describes why this dataset could not be loaded. */
137     public Throwable getException() {
138         return exception;
139     }
140 
141     @Override
142     public String toString() {
143         return id + (ignored ? " [IGNORED]" : "");
144     }
145 
146 }
