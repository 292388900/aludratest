149         runInNewRepo(new GitTest() {
150             @Override
151             public void run(GitClient gitClient) throws Exception {
152                 final String branchName = "branch2";
153                 // there must be a committed file before a branch can be created
154                 createOrOverwriteFile("file1.txt", "content1", true, gitClient);
155                 gitClient.commit(new CommitData("commit1"));
156                 // create the branch
157                 gitClient.createBranch(new BranchCreationData(branchName));
158                 // verify its existence
159                 BranchListData list = new BranchListData();
160                 gitClient.listBranches(list);
161                 assertTrue(branchExists(branchName, gitClient));
162                 // delete the branch
163                 gitClient.deleteBranch(new BranchDeletionData(branchName));
164                 // verify its absence
165                 assertFalse(branchExists(branchName, gitClient));
166             }
167         });
168     }

286         // Skip the test on Mac OSX Yosemite since there is a known issue

291         }

293             @Override

297                 gitClient.commit(new CommitData("initial commit"));
298                 // create test files

302                 // verify the files' absence
303                 StatusData status = getStatus(gitClient);
304                 assertFileContained(UNTRACKED_FILE, status.getUntrackedFiles());
305                 assertFileMissing(ADDED_FILE, status.getUntrackedFiles());
306                 assertFileMissing(ADDED_FILE, status.getAddedFiles());
307                 // stash pop
308                 gitClient.stashPop();
309                 // verify the files' absence
310                 status = getStatus(gitClient);
311                 assertFileContained(UNTRACKED_FILE, status.getUntrackedFiles());
312                 assertFileContained(ADDED_FILE, status.getAddedFiles());
313             }

335                 gitClient.checkout(new CheckoutData("branch2"));

494     private GitClient createGitRepository() {
495         GitClient git = createGitClient();
496         File tempDir = createTempDirectory(git);
497         git.setRelativeWorkingDirectory(new StringData(tempDir.getName()));
498         git.init();
499         return configureUserAndMail(git);
500     }
501 
502     private GitClient configureUserAndMail(GitClient git) {
503         // git config user.name "you"
504         git.config(new ConfigData("user.name", "you"));
505         // git config user.email "you@example.com"
506         git.config(new ConfigData("user.email", "you@example.com"));
507         return git;
508     }

510     private GitClient createAndCloneRepository(GitClient origin, String directory) throws IOException {
511         // create a file "m.txt" and commit master on origin
512         createOrOverwriteFile("m.txt", "m_content", true, origin);
513         origin.commit(new CommitData("initial commit"));
514 
515         // clone origin on local git
516         GitClient copy = createGitClient();
517         copy.setRelativeWorkingDirectory(new StringData(directory));
518         String originPath = workingDirectory(origin).getCanonicalPath();
519         copy.cloneRepository(new CloneRepositoryData(originPath, "."));
520         return copy;
521     }

523     private File createTempDirectory(GitClient git) {
524         String tempRoot = git.getBaseDirectory();
525         File tempDir = new File(tempRoot, UUID.randomUUID().toString());
526         tempDir.mkdirs();
527         return tempDir;
528     }

540     private static File createOrOverwriteFile(String name, String content, boolean add, GitClient gitClient) throws IOException {
541         File file = file(name, gitClient);
542         IOUtil.writeTextFile(file.getAbsolutePath(), content);
543         if (add) {
544             gitClient.add(new AddData(name));
545         }
546         return file;
547     }

549     private static File file(String name, GitClient git) throws IOException {
550         String tempDir = git.getRelativeWorkingDirectory().getValue();
551         return new File(new File(git.getBaseDirectory(), tempDir), name).getCanonicalFile();
552     }

559     private void assertFileContent(String expectedContent, String localFilePath, GitClient git) throws IOException {

564     private String getFileContent(String localFilePath, GitClient git) throws IOException {
565         File file = file(localFilePath, git);
566         String actualContent = getContentOfURI(file.getAbsolutePath(), SystemInfo.getFileEncoding());
567         return actualContent;
568     }

570     public static String getContentOfURI(String uri, String encoding) throws IOException {
571         Reader reader = IOUtil.getReaderForURI(uri, encoding);
572         StringWriter writer = new StringWriter();
573         IOUtil.transfer(reader, writer);
574         IOUtil.close(reader);
575         IOUtil.close(writer);
576         return writer.toString();
577     }

586     private static boolean branchExists(String branchName, GitClient gitClient) {
587         BranchListData list = new BranchListData();
588         gitClient.listBranches(list);
589         for (StringData branch : list.getBranches()) {
590             if (branch.getValue().equals(branchName)) {
591                 return true;
592             }
593         }
594         return false;
595     }

597     private void createTestFilesForReset(GitClient gitClient) throws IOException {
598         // give git-reset a target commit
599         InvocationData data = new InvocationData("git", "commit", "--allow-empty", "-m", "Initial commit.");
600         gitClient.invokeGenerically(data);
601         // create the files
602         createOrOverwriteFile(UNTRACKED_FILE, "untracked", false, gitClient);
603         createOrOverwriteFile(ADDED_FILE, "added", true, gitClient);
604         // verify their presence
605         StatusData status = getStatus(gitClient);
606         assertTrue(containsFile(UNTRACKED_FILE, status.getUntrackedFiles()));
607         assertTrue(containsFile(ADDED_FILE, status.getAddedFiles()));
608     }

614     public void assertFileMissing(String unexpectedFile, List<StringData> actualFiles) {

618     private static boolean containsFile(String fileName, List<StringData> list) {
619         for (StringData file : list) {
620             if (file.getValue().equals(fileName)) {
621                 return true;
622             }
623         }
624         return false;
625     }
