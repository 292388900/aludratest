43      private XmlBasedTestDataProvider createProvider() throws Exception {
44          XmlBasedTestDataProvider provider = new XmlBasedTestDataProvider();
45          AludraTestConfigImpl config = new AludraTestConfigImpl();
46          DefaultConfigurator configurator = new DefaultConfigurator();
47          configurator.configure(config);
48  
49          Map<String, ScriptLibrary> libs = new HashMap<String, ScriptLibrary>();
50          libs.put("default", new DefaultScriptLibrary());
51  
52          ReflectionUtils.setVariableValueInObject(provider, "aludraConfig", config);
53          ReflectionUtils.setVariableValueInObject(provider, "scriptLibraries", libs);
54          return provider;
55      }

57      @Test
58      public void testComplexObjects() throws Exception {
59          XmlBasedTestDataProvider provider = createProvider();
60          List<TestCaseData> testData = provider.getTestDataSets(XmlBasedTestDataProviderTest.class.getDeclaredMethod(
61                  "testMethod1", ComplexData.class, StringData.class));
62          assertEquals(1, testData.size());
63          assertEquals(2, testData.get(0).getData().length);
64  
65          ComplexData cd = (ComplexData) testData.get(0).getData()[0];
66          assertEquals("The Config1", cd.getName());
67          assertNotNull(cd.getSubData());
68          assertEquals("SubData Here.", cd.getSubData().getValue());
69  
70          StringData sd = (StringData) testData.get(0).getData()[1];
71          assertEquals("Some value ending with space ", sd.getValue());
72      }

88      @Test
89      public void testMultiAndNull() throws Exception {
90          XmlBasedTestDataProvider provider = createProvider();
91          List<TestCaseData> testData = provider.getTestDataSets(XmlBasedTestDataProviderTest.class.getDeclaredMethod(
92                  "testMethod2", ComplexData.class, StringData.class));
93          assertEquals(3, testData.size());
94          assertEquals(2, testData.get(0).getData().length);
95          assertEquals(2, testData.get(1).getData().length);
96          assertEquals(2, testData.get(2).getData().length);
97          assertFalse(testData.get(0).isIgnored());
98          assertFalse(testData.get(1).isIgnored());
99          assertTrue(testData.get(2).isIgnored());
100         assertEquals("Ignored for test", testData.get(2).getIgnoredReason());
101 
102         ComplexData cd = (ComplexData) testData.get(0).getData()[0];
103         assertNull(cd.getSubData());
104 
105         cd = (ComplexData) testData.get(1).getData()[0];
106         assertNotNull(cd.getSubData());
107         assertNull(cd.getSubData().getValue());
108 
109         assertEquals("The Config2", cd.getName());
110     }
