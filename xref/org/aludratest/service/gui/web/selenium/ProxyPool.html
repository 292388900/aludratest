16  package org.aludratest.service.gui.web.selenium;
17  
18  import org.aludratest.service.gui.web.selenium.httpproxy.AuthenticatingHttpProxy;
19  import org.aludratest.util.ObjectPool;
20  
21  /**
22   * Creates and manages a pool of {@link AuthenticatingHttpProxy} instances
23   * using an {@link ObjectPool}.
24   * @author Volker Bergmann
25   */
26  public class ProxyPool {
27  
28      /** The {@link ObjectPool} which does the real proxy management */
29      private ObjectPool<AuthenticatingHttpProxy> proxies;
30  
31      /** Creates a pool of proxies which forward all calls to the same target server and port, but each one listening on a different
32       * local port. The used local port numbers begin with 'firstLocalPort' (e.g. 8000) and use the following port numbers (e.g.
33       * 8001, 8002, ...)
34       * @param targetHost the target host
35       * @param targetPortCfg the configured target port
36       * @param firstLocalPort the first local port to be opened by the proxies
37       * @param poolSize The size of the pool. */
38      public ProxyPool(String targetHost, int targetPortCfg, int firstLocalPort, int poolSize) {
39          int targetPort = (targetPortCfg >= 0 ? targetPortCfg : 80);
40          this.proxies = new ObjectPool<AuthenticatingHttpProxy>(poolSize, false);
41          for (int i = 0; i < poolSize; i++) {
42              int localPort = firstLocalPort + i;
43              this.proxies.add(new AuthenticatingHttpProxy(localPort, targetHost, targetPort));
44          }
45      }
46  
47      /** Acquires a proxy from the pool for exclusive use by a single client, waiting if necessary.
48       *  @return a new proxy from the pool
49       *  @throws InterruptedException if the wait has been interrupted */
50      public AuthenticatingHttpProxy acquire() throws InterruptedException {
51          return proxies.acquire();
52      }
53  
54      /** Puts back a used proxy into the pool.
55       *  The client has to call this to put its proxy back to the pool after using it.
56       *  @param proxy The proxy to release*/
57      public void release(AuthenticatingHttpProxy proxy) {
58          proxies.release(proxy);
59      }
60  
61  }
