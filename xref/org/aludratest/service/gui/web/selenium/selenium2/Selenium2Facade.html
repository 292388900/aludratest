108     public Selenium2Facade(SeleniumWrapperConfiguration configuration, String usedSeleniumHost, int usedSeleniumPort)
109             throws MalformedURLException {
110         this.configuration = configuration;
111         this.seleniumUrl = new URL("http://" + usedSeleniumHost + ":" + usedSeleniumPort + "/wd/hub");
112         this.driver = newDriver();
113         this.zIndexSupport = new ZIndexSupport(driver);
114     }

116     /** Creates an instance of the web driver configured in the configuration file. */
117     private WebDriver newDriver() {
118         try {
119             String driverName = configuration.getDriverName();
120             Drivers driverEnum = Drivers.valueOf(driverName);
121 
122             if (configuration.isUsingRemoteDriver()) {
123                 // set Chrome Options capability always, will be ignored for other browsers
124                 DesiredCapabilities caps = new DesiredCapabilities();
125                 if (driverName.toLowerCase(Locale.US).contains("chrome")) {
126                     caps = DesiredCapabilities.chrome();
127                     ChromeOptions opts = new ChromeOptions();
128                     opts.addArguments("--disable-extensions");
129                     caps.setCapability(ChromeOptions.CAPABILITY, opts);
130                 }
131                 else {
132                     caps.setBrowserName(driverEnum.getBrowserName());
133                 }
134                 HttpCommandExecutor executor = new HttpCommandExecutor(seleniumUrl);
135                 return new RemoteWebDriver(executor, caps);
136             }
137             else {
138                 return BeanUtil.newInstance(driverEnum.getDriverClass());
139             }
140         }
141         catch (IllegalStateException e) {
142             throw new TechnicalException("Could not create Web Driver for Selenium", e);
143         }
144     }

165             if (message != null && message.contains("Timed out")) {

175     public void highlight(GUIElementLocator locator) {
176         if (configuration.getHighlightCommands()) {
177             try {
178                 if (this.highlightedElement != null) {
179                     executeScript("arguments[0].style.border='1px solid gray'", highlightedElement);
180                 }
181                 WebElement elementToHighlight = findElement(locator);
182                 executeScript("arguments[0].style.border='3px solid red'", elementToHighlight);
183                 this.highlightedElement = elementToHighlight;
184             } catch (WebDriverException e) {
185                 // It does not matter if highlighting works or not, why a
186                 // possibly thrown exception must be caught to avoid test
187                 // execution termination.
188                 LOGGER.trace("Highlighting failed. ", e);
189             }
190         }
191     }

210     public boolean isInForeground(GUIElementLocator locator) {
211         StaleElementReferenceException exception = null;
212         // try to get z order up to 3 times
213         for (int i = 0; i < MAX_RETRIES_ON_STALE_ELEMENT; i++) {
214             try {
215                 waitUntilPresent(locator);
216 
217                 WebElement element = findElement(locator);
218                 // on success, return the value...
219                 return zIndexSupport.isInForeground(element);
220             } catch (StaleElementReferenceException e) {
221                 // ... on failure, catch and store the exception...
222                 exception = e;
223                 // and repeat (or finish) the loop
224             }
225         }
226         // code has repeatedly failed, so forward the last exception
227         throw exception;
228     }

284     public void selectWindow(WindowLocator locator) {
285         if (locator instanceof TitleLocator) {
286             String requestedTitle = locator.toString();
287 
288             // performance tweak: if the current window is the requested one, return immediately
289             try {
290                 if (requestedTitle.equals(driver.getTitle())) {
291                     return;
292                 }
293             } catch (NoSuchWindowException e) {
294                 // this happens, when trying to call getTitle()
295                 // on a driver which has just close()d the recent window.
296                 // In this case, I fall back to iterating all windows below
297             }
298 
299             // iterate all windows and return the one with the desired title
300             for (String handle : getWindowHandles()) {
301                 String title = driver.switchTo().window(handle).getTitle();
302                 if (requestedTitle.equals(title)) {
303                     return;
304                 }
305             }
306 
307             // if the window has not been found, throw an ElementNotFoundException
308             throw new AutomationException("Element not found");
309         } else {
310             throw ServiceUtil.newUnsupportedLocatorException(locator);
311         }
312     }

319     public String[] getAllWindowTitles() {
320         Map<String, String> handlesAndTitles = getAllWindowHandlesAndTitles();
321         Collection<String> titles = handlesAndTitles.values();
322         LOGGER.debug("getAllWindowTitles() -> {}", titles);
323         return titles.toArray(new String[handlesAndTitles.size()]);
324     }

326     public Map<String, String> getAllWindowHandlesAndTitles() {
327         Map<String, String> handlesAndTitles = new HashMap<String, String>();
328         // store handle and title of current window
329         String initialWindowHandle = driver.getWindowHandle();
330         String title = driver.getTitle();
331         handlesAndTitles.put(initialWindowHandle, title);
332         // iterate all other windows by handle and get their titles
333         String currentHandle = initialWindowHandle;
334         Set<String> handles = getWindowHandles();
335         for (String handle : handles) {
336             if (!handle.equals(initialWindowHandle)) {
337                 LOGGER.debug("Switching to window with handle {}", handle);
338                 driver.switchTo().window(handle);
339                 currentHandle = handle;
340                 handlesAndTitles.put(handle, driver.getTitle());
341                 LOGGER.debug("Window with handle {} has title '{}'", handle, title);
342             }
343         }
344         // switch back to the original window
345         if (!currentHandle.equals(initialWindowHandle)) {
346             driver.switchTo().window(initialWindowHandle);
347         }
348         return handlesAndTitles;
349     }

351     public String[] getAllWindowNames() {
352         String current = driver.getWindowHandle();
353         List<String> names = new ArrayList<String>();
354         for (String handle : getWindowHandles()) {
355             driver.switchTo().window(handle);
356             names.add(executeScript("return window.name").toString());
357         }
358         driver.switchTo().window(current);
359         return names.toArray(new String[names.size()]);
360     }

382     public String captureScreenshotToString() {
383         WebDriver screenshotDriver;
384         if (RemoteWebDriver.class.equals(driver.getClass())) {
385             screenshotDriver = new Augmenter().augment(driver);
386         } else {
387             screenshotDriver = driver;
388         }
389         if (screenshotDriver instanceof TakesScreenshot) {
390             TakesScreenshot tsDriver = (TakesScreenshot) screenshotDriver;
391             return tsDriver.getScreenshotAs(OutputType.BASE64);
392         } else {
393             throw new UnsupportedOperationException(driver.getClass() + " does not implement TakeScreenshot");
394         }
395     }

431     public String getTable(GUIElementLocator locator, int row, int col) {
432         WebElement table = findElement(locator);
433         List<WebElement> tbodies = table.findElements(By.tagName("tbody"));
434         WebElement rowHolder = (tbodies.size() > 0 ? tbodies.get(0) : table);
435         List<WebElement> trs = rowHolder.findElements(By.tagName("tr"));
436         if (row >= trs.size()) {
437             throw new AutomationException("Table cell not found. " +
438                     "Requested row index " + row + " of " + trs.size() + " rows ");
439         }
440         List<WebElement> tds = trs.get(row).findElements(By.tagName("td"));
441         if (col >= tds.size()) {
442             throw new AutomationException("Table cell not found. " +
443                     "Requested column index " + col + " of " + tds.size() + " cells ");
444         }
445         return tds.get(col).getText();
446     }

454     private boolean isEditable(WebElement element) {
455         for (int i = 0; i < MAX_RETRIES_ON_STALE_ELEMENT; i++) {
456             try {
457                 if (!element.isEnabled()) {
458                     return false;
459                 }
460                 String tagName = element.getTagName().toLowerCase();
461                 if (!EDITABLE_ELEMENTS.contains(tagName)) {
462                     return false;
463                 }
464                 if ("input".equals(tagName)) {
465                     String readonly = element.getAttribute("readonly");
466                     return (readonly == null || "false".equals(readonly));
467                 } else {
468                     return true;
469                 }
470             } catch (StaleElementReferenceException e) {
471                 // ignore this and retry in the next loop iteration
472             }
473         }
474         throw new StaleElementReferenceException(element.toString());
475     }

477     private String[] getPropertyValues(GUIElementLocator locator, String propertyName) {
478         WebElement element = findElement(locator);
479         Select select = new Select(element);
480         List<WebElement> options = select.getOptions();
481         ArrayList<String> nonEmptyValues = new ArrayList<String>();
482         for (WebElement option : options) {
483             String value = option.getAttribute(propertyName);
484             if (value != null && value.length() > 0) {
485                 nonEmptyValues.add(value);
486             }
487         }
488         return nonEmptyValues.toArray(new String[nonEmptyValues.size()]);
489     }

531     private void waitFor(ExpectedCondition<?> condition, long timeOutInMillis) {
532         long timeoutInSeconds = (timeOutInMillis + SECOND_MILLIS - 1) / SECOND_MILLIS;
533         int sleepInMillis = configuration.getPauseBetweenRetries();
534         WebDriverWait wait = new WebDriverWait(driver, timeoutInSeconds, sleepInMillis);
535         wait.ignoring(UnknownServerException.class);
536         wait.until(condition);
537     }
