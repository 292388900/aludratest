16  package org.aludratest.service.gui.web.selenium.selenium2;
17  
18  import org.aludratest.service.locator.Locator;
19  import org.aludratest.service.locator.element.CSSLocator;
20  import org.aludratest.service.locator.element.ElementLocators.ElementLocatorsGUI;
21  import org.aludratest.service.locator.element.IdLocator;
22  import org.aludratest.service.locator.element.LabelLocator;
23  import org.aludratest.service.locator.element.XPathLocator;
24  import org.openqa.selenium.By;
25  import org.openqa.selenium.WebDriver;
26  import org.openqa.selenium.WebElement;
27  
28  /**
29   * Maps AludraTest {@link Locator}s to Selenium 2 {@link By} objects.
30   * @author Volker Bergmann
31   */
32  public class LocatorUtil {
33  
34      /** Private constructor of utility class preventing instantiation by other classes */
35      private LocatorUtil() {
36      }
37  
38      public static WebElement findElement(Locator locator, WebDriver driver) {
39          return driver.findElement(by(locator));
40      }
41  
42      public static By by(Locator locator) {
43          if (locator == null) {
44              throw new IllegalArgumentException("Locator is null");
45          }
46          else if (locator instanceof ElementLocatorsGUI) {
47              return new ByElementLocators((ElementLocatorsGUI) locator);
48          } else if (locator instanceof IdLocator) {
49              return By.cssSelector("[id$='" + locator.toString() + "']");
50          } else if (locator instanceof CSSLocator) {
51              return By.cssSelector(locator.toString());
52          } else if (locator instanceof LabelLocator) {
53              return By.linkText(locator.toString());
54          } else if (locator instanceof XPathLocator) {
55              return By.xpath(locator.toString());
56          } else {
57              throw new UnsupportedOperationException("Unsupported locator type: " + locator.getClass().getName());
58          }
59      }
60  
61  }
