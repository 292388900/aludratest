89  /** A full copy of the Selenium HttpCommandExecutor class to be able to use a custom Request timeout.
90   * 
91   * @author falbrech */
92  public class AludraSeleniumHttpCommandExecutor implements CommandExecutor, NeedsLocalLogs {

94      private static final int MAX_REDIRECTS = 10;
95  
96      private static final int THREE_HOURS = (int) TimeUnit.MILLISECONDS.convert(3, TimeUnit.HOURS);

110     /** Constructs a new HttpCommandExecutor for the given remote server.
111      * 
112      * @param addressOfRemoteServer Remove server, or <code>null</code> to fall back to the System property
113      *            <code>webdriver.remote.server</code>, or to <code>http://localhost:4444/wd/hub</code> if system property is not
114      *            set. */
115     public AludraSeleniumHttpCommandExecutor(URL addressOfRemoteServer) {
116         try {
117             remoteServer = addressOfRemoteServer == null ? new URL(System.getProperty("webdriver.remote.server",
118                     "http://localhost:4444/wd/hub")) : addressOfRemoteServer;
119         }
120         catch (MalformedURLException e) {
121             throw new WebDriverException(e);
122         }
123 
124         commandCodec = new JsonHttpCommandCodec();
125         responseCodec = new JsonHttpResponseCodec();
126 
127         synchronized (HttpCommandExecutor.class) {
128             if (httpClientFactory == null) {
129                 httpClientFactory = new HttpClientFactory();
130             }
131         }
132 
133         if (addressOfRemoteServer != null && addressOfRemoteServer.getUserInfo() != null) {
134             // Use HTTP Basic auth
135             UsernamePasswordCredentials credentials = new UsernamePasswordCredentials(addressOfRemoteServer.getUserInfo());
136             client = httpClientFactory.createHttpClient(credentials);
137         }
138         else {
139             client = httpClientFactory.getHttpClient();
140         }
141 
142         // Some machines claim "localhost.localdomain" is the same as "localhost".
143         // This assumption is not always true.
144 
145         String host = remoteServer.getHost().replace(".localdomain", "");
146 
147         targetHost = new HttpHost(host, remoteServer.getPort(), remoteServer.getProtocol());
148     }

150     /** Sets the request timeout to use, in milliseconds. 0 indicates no custom timeout - Selenium defaults apply (3 hours!).
151      * 
152      * @param requestTimeout Request timeout to use, in milliseconds, or 0 to disable custom timeout. */
153     public void setRequestTimeout(int requestTimeout) {
154         this.requestTimeout = requestTimeout;

173     @Override
174     public Response execute(Command command) throws IOException {
175         HttpContext context = new BasicHttpContext();
176 
177         if (command.getSessionId() == null) {
178             if (QUIT.equals(command.getName())) {
179                 return new Response();
180             }
181             if (!GET_ALL_SESSIONS.equals(command.getName()) && !NEW_SESSION.equals(command.getName())) {
182                 throw new SessionNotFoundException("Session ID is null. Using WebDriver after calling quit()?");
183             }
184         }
185 
186         HttpRequest request = commandCodec.encode(command);
187 
188         String requestUrl = remoteServer.toExternalForm().replaceAll("/$", "") + request.getUri();
189 
190         HttpUriRequest httpMethod = createHttpUriRequest(request.getMethod(), requestUrl);
191         for (String name : request.getHeaderNames()) {
192             // Skip content length as it is implicitly set when the message entity is set below.
193             if (!"Content-Length".equalsIgnoreCase(name)) {
194                 for (String value : request.getHeaders(name)) {
195                     httpMethod.addHeader(name, value);
196                 }
197             }
198         }
199 
200         if (httpMethod instanceof HttpPost) {
201             ((HttpPost) httpMethod).setEntity(new ByteArrayEntity(request.getContent()));
202         }
203 
204         if (requestTimeout > 0 && (httpMethod instanceof HttpRequestBase)) {
205             RequestConfig requestConfig = RequestConfig.custom().setConnectionRequestTimeout(15000).setConnectTimeout(15000)
206                     .setSocketTimeout(requestTimeout).build();
207             ((HttpRequestBase) httpMethod).setConfig(requestConfig);
208         }
209         else {
210             // ensure Selenium Standard is set
211             RequestConfig requestConfig = RequestConfig.custom().setConnectionRequestTimeout(60000).setConnectTimeout(60000)
212                     .setSocketTimeout(THREE_HOURS).build();
213             ((HttpRequestBase) httpMethod).setConfig(requestConfig);
214         }
215 
216         try {
217             log(LogType.PROFILER, new HttpProfilerLogEntry(command.getName(), true));
218             HttpResponse response = fallBackExecute(context, httpMethod);
219             log(LogType.PROFILER, new HttpProfilerLogEntry(command.getName(), false));
220 
221             response = followRedirects(client, context, response, /* redirect count */0);
222 
223             return createResponse(response, context);
224         }
225         catch (UnsupportedCommandException e) {
226             if (e.getMessage() == null || "".equals(e.getMessage())) {
227                 throw new UnsupportedOperationException("No information from server. Command name was: " + command.getName(),
228                         e.getCause());
229             }
230             throw e;
231         }
232         catch (SocketTimeoutException e) {
233             LoggerFactory.getLogger(AludraSeleniumHttpCommandExecutor.class).warn(
234                     "Timeout in HTTP Command Executor. Timeout was "
235                             + ((HttpRequestBase) httpMethod).getConfig().getSocketTimeout());
236             throw e;
237         }
238     }

240     private static HttpUriRequest createHttpUriRequest(HttpMethod method, String url) {
241         switch (method) {
242             case DELETE:
243                 return new HttpDelete(url);
244             case GET:
245                 return new HttpGet(url);
246             case POST:
247                 return new HttpPost(url);
248         }
249         throw new AssertionError("Unsupported method: " + method);
250     }
251 
252     private HttpResponse fallBackExecute(HttpContext context, HttpUriRequest httpMethod) throws IOException {
253         try {
254             return client.execute(targetHost, httpMethod, context);
255         }
256         catch (BindException e) {
257             // If we get this, there's a chance we've used all the local ephemeral sockets
258             // Sleep for a bit to let the OS reclaim them, then try the request again.
259             try {
260                 Thread.sleep(2000);
261             }
262             catch (InterruptedException ie) {
263                 throw Throwables.propagate(ie);
264             }
265         }
266         catch (NoHttpResponseException e) {
267             // If we get this, there's a chance we've used all the remote ephemeral sockets
268             // Sleep for a bit to let the OS reclaim them, then try the request again.
269             try {
270                 Thread.sleep(2000);
271             }
272             catch (InterruptedException ie) {
273                 throw Throwables.propagate(ie);
274             }
275         }
276         return client.execute(targetHost, httpMethod, context);
277     }

279     private HttpResponse followRedirects(HttpClient client, HttpContext context, HttpResponse response, int redirectCount) {
280         if (!isRedirect(response)) {
281             return response;
282         }
283 
284         try {
285             // Make sure that the previous connection is freed.
286             HttpEntity httpEntity = response.getEntity();
287             if (httpEntity != null) {
288                 EntityUtils.consume(httpEntity);
289             }
290         }
291         catch (IOException e) {
292             throw new WebDriverException(e);
293         }
294 
295         if (redirectCount > MAX_REDIRECTS) {
296             throw new WebDriverException("Maximum number of redirects exceeded. Aborting");
297         }
298 
299         String location = response.getFirstHeader("location").getValue();
300         URI uri;
301         try {
302             uri = buildUri(context, location);
303 
304             HttpGet get = new HttpGet(uri);
305             get.setHeader("Accept", "application/json; charset=utf-8");
306             HttpResponse newResponse = client.execute(targetHost, get, context);
307             return followRedirects(client, context, newResponse, redirectCount + 1);
308         }
309         catch (URISyntaxException e) {
310             throw new WebDriverException(e);
311         }
312         catch (ClientProtocolException e) {
313             throw new WebDriverException(e);
314         }
315         catch (IOException e) {
316             throw new WebDriverException(e);
317         }
318     }
319 
320     private URI buildUri(HttpContext context, String location) throws URISyntaxException {
321         URI uri;
322         uri = new URI(location);
323         if (!uri.isAbsolute()) {
324             HttpHost host = (HttpHost) context.getAttribute(HttpCoreContext.HTTP_TARGET_HOST);
325             uri = new URI(host.toURI() + location);
326         }
327         return uri;
328     }

330     private boolean isRedirect(HttpResponse response) {
331         int code = response.getStatusLine().getStatusCode();
332 
333         return (code == 301 || code == 302 || code == 303 || code == 307) && response.containsHeader("location");
334     }
335 
336     private Response createResponse(HttpResponse httpResponse, HttpContext context) throws IOException {
337         org.openqa.selenium.remote.http.HttpResponse internalResponse = new org.openqa.selenium.remote.http.HttpResponse();
338 
339         internalResponse.setStatus(httpResponse.getStatusLine().getStatusCode());
340         for (Header header : httpResponse.getAllHeaders()) {
341             for (HeaderElement headerElement : header.getElements()) {
342                 internalResponse.addHeader(header.getName(), headerElement.getValue());
343             }
344         }
345 
346         HttpEntity entity = httpResponse.getEntity();
347         if (entity != null) {
348             try {
349                 internalResponse.setContent(EntityUtils.toByteArray(entity));
350             }
351             finally {
352                 EntityUtils.consume(entity);
353             }
354         }
355 
356         Response response = responseCodec.decode(internalResponse);
357         if (response.getSessionId() == null) {
358             HttpHost finalHost = (HttpHost) context.getAttribute(HttpCoreContext.HTTP_TARGET_HOST);
359             String uri = finalHost.toURI();
360             String sessionId = HttpSessionId.getSessionId(uri);
361             response.setSessionId(sessionId);
362         }
363 
364         return response;
365     }
