16  package org.aludratest.service.gui.web.selenium.selenium2.condition;
17  
18  import java.text.MessageFormat;
19  
20  import org.aludratest.service.gui.web.selenium.selenium2.LocatorSupport;
21  import org.aludratest.service.locator.element.XPathLocator;
22  import org.openqa.selenium.By;
23  import org.openqa.selenium.InvalidSelectorException;
24  import org.openqa.selenium.JavascriptExecutor;
25  import org.openqa.selenium.WebDriver;
26  import org.openqa.selenium.WebDriverException;
27  import org.openqa.selenium.WebElement;
28  import org.slf4j.Logger;
29  import org.slf4j.LoggerFactory;
30  
31  /** Performs z-index calculations with Selenium 2.
32   * @author Marcel Malitz
33   * @author Volker Bergmann */
34  public class ZIndexSupport {
35  
36      private static final Logger LOGGER = LoggerFactory.getLogger(ZIndexSupport.class);
37  
38      private static final int DEFAULT_Z_INDEX = 0;
39  
40      private static final String Z_INDEX_SEARCH_XPATH = "//iframe[contains(@style, ''z-index'')])[{0}]";
41  
42      private final LocatorSupport locatorSupport;
43      private final WebDriver driver;
44  
45      // constructor -------------------------------------------------------------
46  
47      /** Constructor.
48       * @param locatorSupport */
49      public ZIndexSupport(LocatorSupport locatorSupport) {
50          this.locatorSupport = locatorSupport;
51          this.driver = locatorSupport.getDriver();
52      }
53  
54      // properties --------------------------------------------------------------
55  
56      /** @return the {@link #locatorSupport} */
57      public LocatorSupport getLocatorSupport() {
58          return locatorSupport;
59      }
60  
61      // operational interface ---------------------------------------------------
62  
63      /** Checks if a element is blocked by a modal dialog.
64       * @param element the element to be checked
65       * @return true if the element is in the foreground, false if the element is in the background or absent */
66      public boolean isInForeground(WebElement element) {
67          return (getCurrentZIndex(element) >= getMaxZIndex());
68      }
69  
70      // implementation ----------------------------------------------------------
71  
72      /** To get the z-index (defined in the attribute "style") for the operated element. There are 3 possibilities for retrieving
73       * z-index: <br/>
74       * <ol>
75       * <li>If a z-index is defined for this element or its ancestor, then return this value</li>
76       * <li>If no z-index is defined for this element and its ancestor, then use the base z-index for this page</li>
77       * <li>For an element of the type "LabelLocator", the base z-index will be returned</li>
78       * </ol>
79       * @param element The element to check.
80       * @return current z-Index */
81      private int getCurrentZIndex(WebElement element) {
82          String zIndex = null;
83          try {
84              do {
85                  zIndex = element.getCssValue("z-index");
86                  element = locatorSupport.getParent(element);
87              } while ("auto".equals(zIndex) && element != null);
88          } catch (InvalidSelectorException e) {
89              // this occurs when having reached the root element
90          }
91          catch (WebDriverException e) {
92              // for PhantomJS, such an exception is thrown instead
93              if (e.getMessage() == null || !e.getMessage().contains(" should be an element")) {
94                  throw e;
95              }
96              // otherwise OK
97          }
98          int value = parseZIndex(zIndex);
99          LOGGER.trace("WebElement {} has z index {}", element, value);
100         return value;
101     }
102 
103     /** To get the biggest value of z-index for all of the elements on current page. The element with the biggest value of z-index
104      * will be shown in foreground. The elements with the lower value of z-index will be shown in background.
105      * @return the biggest value of z-index on current page */
106     private int getMaxZIndex() {
107         int zIndex = getBaseZIndex();
108         int zIndexCount = 0;
109         zIndexCount = getZIndexCount();
110         for (int i = 0; i < zIndexCount; i++) {
111             int tmpzIndex = getzIndex(i + 1);
112             zIndex = (tmpzIndex > zIndex) ? tmpzIndex : zIndex;
113         }
114         return zIndex;
115     }
116 
117     private int getBaseZIndex() {
118         // If it has a default z-Index defined in code, then get its value
119         if (getHistoryFrameCount() > 0) {
120             return getzIndex(0);
121             // If it has not defined a default z-Index in code, then set it to a default value
122         } else {
123             return DEFAULT_Z_INDEX;
124         }
125     }
126 
127     private int getZIndexCount() {
128         return getXPathCount("//iframe[contains(@style, \"z-index\")]");
129     }
130 
131     private int getHistoryFrameCount() {
132         int historyFrameCount = 0;
133         historyFrameCount = getXPathCount("//iframe[starts-with(@id, \"history-frame\")]");
134         return historyFrameCount;
135     }
136 
137     private int getzIndex(int index) {
138         int tmpzIndex = DEFAULT_Z_INDEX;
139         // If a base value is defined in code, it will overwrite the default value
140         try {
141             String zIndexSearchXPath = MessageFormat.format(Z_INDEX_SEARCH_XPATH, index);
142             WebElement element = locatorSupport.findElementImmediately(new XPathLocator(zIndexSearchXPath));
143             String tmpElement = (String) executeScript(element.getAttribute("style"), driver, index);
144             tmpzIndex = getzIndexFromStyle(tmpElement);
145         } catch (InvalidSelectorException e) {
146             // This may happen for some elements and needs to be ignored
147         }
148         return tmpzIndex;
149     }
150 
151     private int getzIndexFromStyle(String style) {
152         if (style.endsWith("undefined")) {
153             // a normal element without z-Index defined
154             return getBaseZIndex();
155         } else {
156             // an element with z-Index
157             for (String tmp : style.split(";")) {
158                 if (tmp.trim().startsWith("z-index")) {
159                     return Integer.parseInt(tmp.replace("z-index:", "").trim());
160                 }
161             }
162         }
163         return getBaseZIndex();
164     }
165 
166     private int parseZIndex(String zIndexString) {
167         if ("auto".equals(zIndexString)) {
168             return getBaseZIndex();
169         } else {
170             return Integer.parseInt(zIndexString.trim());
171         }
172     }
173 
174     private int getXPathCount(String xpath) {
175         return driver.findElements(By.xpath(xpath)).size();
176     }
177 
178     private Object executeScript(String script, Object... arguments) {
179         return ((JavascriptExecutor) driver).executeScript(script, arguments);
180     }
181 
182 }
