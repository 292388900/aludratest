185     /** Returns the implementor class for the given component class. Subclasses can override to specify own implementation classes,
186      * but you <b>must</b> also override {@link #getRoleHint()}.
187      * 
188      * @param componentClass Class of the component, e.g. <code>Button.class</code>.
189      * 
190      * @return The implementing component class, or <code>null</code> if no implementor class is available for the given component
191      *         class. Do <b>not</b> throw any exception; the calling method will deal with this. */
192     protected <T extends GUIComponent> Class<? extends T> getImplementorClass(Class<T> componentClass) {
193         @SuppressWarnings("unchecked")
194         Class<? extends T> implClass = (Class<? extends T>) componentImplClasses.get(componentClass);
195         return implClass;
196     }
197 
198     /** Returns the role hint which is used by this component factory. If you want to register your own implementations of the
199      * component classes, you also have to override this method to avoid name clashes in the internal Plexus component registry.
200      * 
201      * @return The role hint to use for this component factory. */
202     protected String getRoleHint() {
203         return "default";
204     }

206     /** Configures the given freshly instantiated component to use the given identifiers. Subclasses can add more configuration
207      * logic, but should always invoke the super implementation.
208      * 
209      * @param component Component to configure.
210      * @param locator Locator to set in the component.
211      * @param componentClass Component class identifying the type of the component, e.g. <code>Button.class</code>.
212      * @param elementName Element name which was set in the factory method call or automatically determined if possible, or
213      *            <code>null</code>. */
214     protected void configureComponent(GUIComponent component, Locator locator, Class<?> componentClass, String elementName) {
215         if (component instanceof AbstractGUIComponent) {
216             ((AbstractGUIComponent) component).configure(aludraGUI, locator, componentClass.getSimpleName(), elementName);
217         }
218     }

220     protected final <T extends GUIComponent> T createComponent(Class<T> componentClass, Locator locator) {
221         return createComponent(componentClass, locator, determineDefaultElementName());
222     }
223 
224     protected final <T extends GUIComponent> T createComponent(Class<T> componentClass, Locator locator, String elementName) {
225         Class<? extends T> implClass = getImplementorClass(componentClass);
226         if (implClass == null) {
227             throw new TechnicalException("No implementation class found for component class " + componentClass.getName());
228         }
229 
230         try {
231             T component;
232             synchronized (plexusContainer) {
233                 // check if Plexus container already "knows" this component class
234                 if (plexusContainer.getComponentDescriptor(componentClass.getName(), getRoleHint()) == null) {
235                     registerComponentDescriptor(componentClass, implClass);
236                 }
237 
238                 component = plexusContainer.lookup(componentClass, getRoleHint());
239             }
240             configureComponent(component, locator, componentClass, elementName);
241             return component;
242         }
243         catch (ComponentLookupException e) {
244             throw new TechnicalException("Could not create component of type " + componentClass.getSimpleName(), e);
245         }
246         catch (CycleDetectedInComponentGraphException e) {
247             throw new TechnicalException("Could not register component implementation of type " + componentClass.getSimpleName(),
248                     e);
249         }
250     }

252     private <T> void registerComponentDescriptor(Class<T> componentClass, Class<? extends T> implClass)
253             throws CycleDetectedInComponentGraphException {
254         if (!componentClass.isAssignableFrom(implClass)) {
255             throw new RuntimeException("Class " + implClass.getName() + " does not implement interface "
256                     + componentClass.getName());
257         }
258 
259         ComponentDescriptor<T> desc = new ComponentDescriptor<T>();
260         desc.setRole(componentClass.getName());
261         desc.setRoleClass(componentClass);
262         desc.setRoleHint(getRoleHint());
263         desc.setImplementationClass(implClass);
264         desc.setIsolatedRealm(false);
265         desc.setInstantiationStrategy("per-lookup");
266 
267         ClassRealm realm = plexusContainer.getLookupRealm();
268         if (realm == null) {
269             realm = plexusContainer.getContainerRealm();
270         }
271         desc.setRealm(realm);
272 
273         // lookup Requirements in class and parent classes
274         Class<?> clazz = implClass;
275         while (clazz != null && clazz != Object.class) {
276             for (Field f : implClass.getDeclaredFields()) {
277                 Requirement req = f.getAnnotation(Requirement.class);
278                 if (req != null) {
279                     desc.addRequirement(createComponentRequirement(f, req));
280                 }
281 
282             }
283             clazz = clazz.getSuperclass();
284         }
285 
286         plexusContainer.addComponentDescriptor(desc);
287     }

289     private static ComponentRequirement createComponentRequirement(Field field, Requirement requirement) {
290         ComponentRequirement cr = new ComponentRequirement();
291         cr.setFieldName(field.getName());
292         cr.setOptional(requirement.optional());
293         if (requirement.role() != null && requirement.role() != Object.class) {
294             cr.setRole(requirement.role().getName());
295         }
296         else {
297             cr.setRole(field.getType().getName());
298         }
299         if (requirement.hint() != null) {
300             cr.setRoleHint(requirement.hint());
301         }
302 
303         return cr;
304     }

306     private static String determineDefaultElementName() {
307         StackTraceElement[] stackTrace = Thread.currentThread().getStackTrace();
308         String lastMethodName = null;
309 
310         for (StackTraceElement ste : stackTrace) {
311             String className = ste.getClassName();
312             if (className != null && className.contains("UIMap")) {
313                 lastMethodName = ste.getMethodName();
314             }
315         }
316 
317         // getCloseButton() => closeButton
318         if (lastMethodName != null && lastMethodName.startsWith("get") && lastMethodName.length() > 3) {
319             char ch = lastMethodName.charAt(3);
320             // could also be "getterButton()", so only remove when uppercase follows
321             if (Character.isUpperCase(ch)) {
322                 lastMethodName = Character.toLowerCase(ch) + lastMethodName.substring(4);
323             }
324         }
325 
326         return lastMethodName;
327     }
