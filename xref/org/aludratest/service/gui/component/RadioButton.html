16  package org.aludratest.service.gui.component;
17  
18  import org.aludratest.service.gui.AludraGUI;
19  import org.aludratest.service.locator.Locator;
20  import org.aludratest.util.data.helper.DataMarkerCheck;
21  
22  /**
23   * Represents a radio button on a GUI.
24   * @author Joerg Langnickel
25   * @author Volker Bergmann
26   */
27  public class RadioButton extends InputComponent<RadioButton> {
28  
29      /**
30       * Constructor.
31       * @param aludraGui the underlying {@link AludraGUI} service instance
32       * @param locator a locator for the referenced radio button
33       */
34      public RadioButton(AludraGUI aludraGui, Locator locator) {
35          super(aludraGui, locator);
36      }
37  
38      /** Constructor.
39       *  @param aludraGui the underlying {@link AludraGUI} service instance
40       *  @param locator a locator for the referenced element
41       *  @param elementName an explicit name to use for the component */
42      public RadioButton(AludraGUI aludraGui, Locator locator, String elementName) {
43          super(aludraGui, locator, elementName);
44      }
45      
46      /** Selects this radio button. */
47      public void select() {
48          perform().selectRadiobutton(elementType, elementName, locator, taskCompletionTimeout);
49      }
50  
51      /** Selects this radio button if and only if the passed string parameter has the value <code>"true"</code>. In every other
52       * case, no action is performed.
53       * 
54       * @param value String parameter to indicate if a select operation shall be performed on this radio button. */
55      public void select(String value) {
56          if (!DataMarkerCheck.isNull(value)) {
57              if (Boolean.parseBoolean(value)) {
58                  select();
59              }
60          }
61      }
62  
63      /** Asserts that the radio button is checked */
64      public void assertChecked() {
65          verify().assertChecked(elementType, elementName, locator);
66      }
67  
68      /** Asserts that this Radio button is in the expected state, passed by expected string. If the expected string is null or
69       * marked as null, no operation will be executed.
70       * 
71       * @param expected <code>"true"</code> or <code>"false"</code>, or <code>null</code> or marked as null to not perform any
72       *            assertion. */
73      public void assertChecked(String expected) {
74          if (!DataMarkerCheck.isNull(expected)) {
75              verify().assertChecked(elementType, elementName, Boolean.parseBoolean(expected), locator);
76          }
77      }
78  
79  }
