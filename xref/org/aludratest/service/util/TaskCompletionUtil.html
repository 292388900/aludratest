16  package org.aludratest.service.util;
17  
18  import org.aludratest.exception.PerformanceFailure;
19  import org.aludratest.service.SystemConnector;
20  import org.slf4j.Logger;
21  import org.slf4j.LoggerFactory;
22  
23  /**
24   * Provides utility methods for controlling the completion of (synchronous or asynchronous) tasks.
25   * @author Volker Bergmann
26   */
27  public class TaskCompletionUtil {
28  
29      private static final Logger LOGGER = LoggerFactory.getLogger(TaskCompletionUtil.class);
30  
31      private TaskCompletionUtil() { }
32  
33      /**
34       * Waits a limited amount of time for an activity to start and finish
35       * using the {@link SystemConnector#isBusy()} of the connector object.
36       * First a startTimeout period is spent waiting for an activity to occur.
37       * If no activity was noticed, the method returns false immediately.
38       * Otherwise the method waits for the period of the completionTimeout
39       * until no activity is reported any more. If the activity is noticed
40       * to be stopped, the method returns true immediately.
41       * If the system is still busy after the timeout, a {@link PerformanceFailure}
42       * is thrown.
43       * @param connector the {@link SystemConnector} for controlling the system
44       * @param failureMessage the exception message text to use in case of a timeout
45       * @param startTimeout the initial period to wait in milliseconds until activity is observed
46       * @param completionTimeout the maximum time to wait milliseconds for task completion (including the startTimeout)
47       * @param pollingInterval the interval to wait between checking the activity status in milliseconds
48       * @return true if an activity was observed, otherwise false
49       * @throws PerformanceFailure if the completionTimeout was exceeded before activity ended
50       */
51      public static boolean waitForActivityAndCompletion(SystemConnector connector, final String failureMessage,
52              int startTimeout, int completionTimeout, int pollingInterval) {
53          long startTime = System.currentTimeMillis();
54          if (!waitUntilBusy(connector, startTime + startTimeout, startTimeout, pollingInterval)) {
55              return false;
56          }
57          int dt = (int) (System.currentTimeMillis() - startTime);
58          // a task is running, I wait until it is finished or I get impatient...
59          LOGGER.debug("Activity was observed on {}, now waiting for completion " +
60                  "until the completionTimeout of {} ms", connector, completionTimeout);
61          waitUntilNotBusy(connector, completionTimeout - dt, pollingInterval, failureMessage);
62          LOGGER.debug("{} finished its activity", connector);
63          return true;
64      }
65  
66      /**
67       * Repeats a delayed loop until the provided {@link SystemConnector} reports
68       * false on invocation of {@link SystemConnector#isBusy()}
69       * 
70       * @param connector
71       *            the {@link SystemConnector} to query if the system is busy
72       * @param timeout
73       *            the maximum number of milliseconds to wait
74       * @param pollingInterval
75       *            the number of milliseconds to wait between system state
76       *            queries
77       * @param failureMessage
78       *            The failure message to emit if the system is still busy after
79       *            timeout
80       */
81      public static void waitUntilNotBusy(SystemConnector connector, int timeout, int pollingInterval, String failureMessage) {
82          long timeoutTime = System.currentTimeMillis() + timeout;
83          boolean busy;
84          long time;
85          do {
86              busy = isBusy(connector);
87              LOGGER.debug("{} is {}", connector, (busy ? "busy" : "not busy"));
88              if (busy) {
89                  sleep(pollingInterval);
90              }
91              time = System.currentTimeMillis();
92          } while (busy && time < timeoutTime);
93          if (busy) {
94              // timeout is exceeded and the task is still busy,
95              // so I throw an appropriate exception
96              LOGGER.error("{} did not finish its activities within the completionTimeout period", connector);
97              throw new PerformanceFailure(failureMessage);
98          }
99      }
100 
101     private static boolean waitUntilBusy(SystemConnector connector, long timeoutTime, int timeoutInterval, int pollingInterval) {
102         LOGGER.debug("Waiting for activity on {}", connector);
103         long time;
104         boolean busy;
105         do {
106             busy = isBusy(connector);
107             LOGGER.debug("{} is {}", connector, (busy ? "busy" : "not busy"));
108             if (!busy) {
109                 sleep(pollingInterval);
110             }
111             time = System.currentTimeMillis();
112         } while (!busy && time < timeoutTime);
113         if (!busy) {
114             // I probably missed the task execution and better stop here
115             LOGGER.debug("No activity observed within the timeout of {} ms", timeoutInterval);
116             return false;
117         }
118         return true;
119     }
120 
121     private static boolean isBusy(SystemConnector connector) {
122         if (connector == null) {
123             return false;
124         }
125         SystemBusyIndicator busyIndicator = connector.getConnector(SystemBusyIndicator.class);
126         return busyIndicator != null ? busyIndicator.isBusy() : false;
127     }
128 
129     private static void sleep(int duration) {
130         try {
131             Thread.sleep(duration);
132         } catch (InterruptedException e) {
133             LOGGER.error("Thread.sleep() was interrupted. ", e);
134         }
135     }
136 
137 }
