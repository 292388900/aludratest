16  package org.aludratest.service.cmdline;
17  
18  import org.aludratest.util.data.StringData;
19  import org.aludratest.util.validator.ContainsValidator;
20  import org.aludratest.util.validator.EqualsValidator;
21  import org.databene.commons.Validator;
22  import org.databene.commons.validator.PrefixValidator;
23  import org.databene.commons.validator.SuffixValidator;
24  
25  /** Class for verifying process output line-wise.
26   * @author Volker Bergmann */
27  public class Line {
28  
29      private final Out out;
30  
31      Line(Out out) {
32          this.out = out;
33      }
34  
35      /** Asserts that a text line is identical to the provided string.
36       * @param text the expected line content
37       * @return this */
38      public Line assertEquals(StringData text) {
39          return assertMatch(new EqualsValidator(text.getValue()));
40      }
41  
42      /** Asserts that a text line starts with the given prefix.
43       * @param prefix the expected prefix of the line
44       * @return this */
45      public Line assertPrefix(StringData prefix) {
46          return assertMatch(new PrefixValidator(prefix.getValue()));
47      }
48  
49      /** Asserts that a text line ends with the given suffix.
50       * @param suffix the expected suffix of the line
51       * @return this */
52      public Line assertSuffix(StringData suffix) {
53          return assertMatch(new SuffixValidator(suffix.getValue()));
54      }
55  
56      /** Asserts that a text line contains the given text.
57       * @param expectedText the txt expected to be contained in the line
58       * @return this */
59      public Line assertContains(StringData expectedText) {
60          return assertMatch(new ContainsValidator(expectedText.getValue()));
61      }
62  
63      protected Line assertMatch(Validator<String> validator) {
64          out.assertNextLineMatches(validator);
65          return this;
66      }
67  
68  }
