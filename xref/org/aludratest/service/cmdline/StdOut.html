16  package org.aludratest.service.cmdline;
17  
18  import java.io.OutputStream;
19  
20  import org.aludratest.exception.FunctionalFailure;
21  import org.databene.commons.Validator;
22  
23  /** Provides access to the standard output of a process.
24   * @author Volker Bergmann */
25  public class StdOut extends Out {
26  
27      protected StdOut(CommandLineProcess<?> process) {
28          super(process);
29      }
30  
31      /** Redirects the output to a stream.
32       * @param out the target stream
33       * @return this */
34      @Override
35      public Out redirectTo(OutputStream out) {
36          process.redirectStdOutTo(out);
37          return this;
38      }
39  
40      /** Asserts that the validator matches the next line of the output
41       * @param validator the matcher
42       * @return this */
43      @Override
44      public Out assertNextLineMatches(Validator<String> validator) {
45          process.assertNextLineOfStdOutMatches(validator);
46          return this;
47      }
48  
49      /** Skips the output lines until a line matches the validator or the end of the file is reached.
50       * @param validator the matcher
51       * @return a Line object for the first matching line */
52      @Override
53      public Line skipUntilLineMatches(Validator<String> validator) {
54          process.skipStdOutUntilLineMatches(validator);
55          return new Line(this);
56      }
57  
58      /** Asserts the output to be empty.
59       * @return this
60       * @throws a {@link FunctionalFailure} if output is found */
61      @Override
62      public Out assertEmpty() {
63          process.assertEmptyStdOut();
64          return this;
65      }
66  
67  }
