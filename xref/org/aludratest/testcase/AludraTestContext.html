16  package org.aludratest.testcase;
17  
18  import org.aludratest.service.AludraContext;
19  import org.aludratest.service.ComponentId;
20  import org.aludratest.testcase.event.TestStepInfo;
21  
22  /** Test context provided to AludraTestCase instances in order to access information necessary for initialization and framework
23   * interaction.
24   * @author Volker Bergmann
25   * @author falbrech */
26  public interface AludraTestContext extends AludraContext {
27  
28      /** CLoses all services which were opened by this context. */
29      public void closeServices();
30  
31      /** Closes a service.
32       * @param serviceId the {@link ComponentId} of the service to close */
33      public void closeService(ComponentId<?> serviceId);
34  
35      /** Starts a new group of test steps with the given name.
36       * @param name of the group to create */
37      public void newTestStepGroup(String name);
38  
39      /** Reports an execution failure to the framework.
40       * @param errorMessage the error message assigned to the failure.
41       * @param status the {@link TestStatus} indicating the failure. <code>status.isFailure()</code> should be <code>true</code>. */
42      public void logError(String errorMessage, TestStatus status);
43  
44      /** Reports an execution error (a kind of "uncaught" exception) to the framework.
45       * @param errorMessage the error message assigned to the error. Can be the message of the exception.
46       * @param t the {@link Throwable} to be reported. */
47      public void logError(String errorMessage, Throwable t);
48  
49      /** Fires a new test step to all listeners of the current execution context.
50       * 
51       * @param step Step to notify listeners about. */
52      public void fireTestStep(TestStepInfo step);
53  
54  }
