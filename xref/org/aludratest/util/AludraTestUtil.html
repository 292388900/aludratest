40      /** A {@link Map} that provides null value replacements
41       *  for primitive data types and their wrapper classes. */
42      private static final Map<Class<?>, Object> NULL_REPLACEMENTS;
43  
44      static {
45          NULL_REPLACEMENTS = new HashMap<Class<?>, Object>();
46          NULL_REPLACEMENTS.put(boolean.class, false);
47          NULL_REPLACEMENTS.put(byte.class, (byte) 0);
48          NULL_REPLACEMENTS.put(short.class, (short) 0);
49          NULL_REPLACEMENTS.put(int.class, 0);
50          NULL_REPLACEMENTS.put(long.class, (long) 0);
51          NULL_REPLACEMENTS.put(float.class, 0f);
52          NULL_REPLACEMENTS.put(double.class, 0.);
53          NULL_REPLACEMENTS.put(char.class, (char) 0);
54          NULL_REPLACEMENTS.put(Boolean.class, false);
55          NULL_REPLACEMENTS.put(Byte.class, (byte) 0);
56          NULL_REPLACEMENTS.put(Short.class, (short) 0);
57          NULL_REPLACEMENTS.put(Integer.class, 0);
58          NULL_REPLACEMENTS.put(Long.class, (long) 0);
59          NULL_REPLACEMENTS.put(Float.class, 0f);
60          NULL_REPLACEMENTS.put(Double.class, 0.);
61          NULL_REPLACEMENTS.put(Character.class, (char) 0);
62          NULL_REPLACEMENTS.put(String.class, "");
63      }
64  
65      /** Private constructor of utility class preventing instantiation by other classes */
66      private AludraTestUtil() {
67      }
68  
69      /** Returns a value of the given type which can be used to replace null values. This is useful to provide a non-null value from
70       * generic invocations on methods which are expected to return a non-null value (typically a primitive data type). Main
71       * purpose is its use in the {@link ControlFlowHandler} (proxy), which is allowed to skip actual method execution but has to
72       * return a value that appears valid to the caller.
73       * @param type
74       * @return null for classes, an 'empty' or zero-like value for primitive return types */
75      public static Object nullOrPrimitiveDefault(Class<?> type) {
76          return NULL_REPLACEMENTS.get(type);
77      }
78  
79      /**
80       * Wraps the given object with a dynamic proxy that implements a parent interface of the object
81       * and transparently add control flow logic using the {@link ControlFlowHandler}.
82       */
83      public static <T, U extends T> T wrapWithControlFlowHandler(U object, Class<T> interfaceType,
84              ComponentId<? extends AludraService> serviceId,
85              SystemConnector systemConnector, AludraContext context) {
86          AludraTestConfig config = context.newComponentInstance(AludraTestConfig.class);
87          boolean stopOnException;
88          if (Interaction.class.isAssignableFrom(interfaceType)) {
89              stopOnException = config.isStopTestCaseOnInteractionException();
90          } else if (Verification.class.isAssignableFrom(interfaceType)) {
91              stopOnException = config.isStopTestCaseOnVerificationException();
92          } else {
93              stopOnException = config.isStopTestCaseOnOtherException();
94          }
95          if (context instanceof AludraTestContext) {
96              InvocationHandler invocationHandler = new ControlFlowHandler(object, serviceId, systemConnector,
97                      (AludraTestContext) context, stopOnException, !Condition.class.isAssignableFrom(interfaceType));
98              return AludraTestUtil.<T, U> wrapWithInvocationHandler(interfaceType, invocationHandler);
99          }
100         return object;
101     }
102 
103     /**
104      * Calls the Java dynamic proxy API to dynamically implement the given interfaceType
105      * using the given invocationHandler.
106      * @see InvocationHandler
107      */
108     @SuppressWarnings("unchecked")
109     public static <T, U extends T> U wrapWithInvocationHandler(Class<T> interfaceType, InvocationHandler invocationHandler) {
110         try {
111             ClassLoader classLoader = Thread.currentThread().getContextClassLoader();
112             return (U) Proxy.newProxyInstance(classLoader, new Class[] { interfaceType }, invocationHandler);
113         } catch (Exception e) {
114             throw new RuntimeException(e);
115         }
116     }
117 
118     /** If the argument is an {@link InvocationTargetException}
119      *  with a non-null cause, the method returns that cause exception,
120      *  otherwise the {@link Throwable} object itself. */
121     public static Throwable unwrapInvocationTargetException(Throwable t) {
122         while (t instanceof InvocationTargetException && t.getCause() != null) {
123             t = t.getCause();
124         }
125         return t;
126     }
127 
128     public static StackTraceElement getStackTraceElement(int levelsUp) {
129         StackTraceElement[] stackTrace = Thread.currentThread().getStackTrace();
130         int index = levelsUp + 1;
131         if (index >= stackTrace.length) {
132             return null;
133         }
134         return stackTrace[index];
135     }
136 
137     public static TestStatus getTestStatus(Throwable t) {
138         if (t instanceof AludraTestException) {
139             return ((AludraTestException) t).getTestStatus();
140         } else {
141             return TestStatus.INCONCLUSIVE;
142         }
143     }
