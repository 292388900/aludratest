16  package org.aludratest.util.retry;
17  
18  import java.util.concurrent.Callable;
19  
20  import org.aludratest.util.AludraTestUtil;
21  import org.databene.commons.Assert;
22  
23  /** Invokes a method on an object with the provided arguments. If an exception of a tolerated exception type occurs, the the
24   * invocation is repeated until
25   * <ul>
26   * <li>the invocation is successful</li>
27   * <li>an exception of an untolerated type occurs</li>
28   * <li>the maximum number of allowed retries is exceeded</li>
29   * </ul>
30   * @author Volker Bergmann */
31  public class RetryService {
32  
33      /** Performs the invocation an exception handling as described in the class doc.
34       * @param target
35       * @param toleratedThrowable
36       * @param maxRetries
37       * @return
38       * @throws Throwable */
39      public static <T> T call(Callable<T> target, Class<? extends Throwable> toleratedThrowable, int maxRetries)
40              throws Throwable { // NOSONAR
41          // check preconditions
42          Assert.notNull(target, "target");
43  
44          // invocation cycle
45          int retryCount = 0;
46          boolean doRetry;
47          Throwable recentException = null;
48          do {
49              doRetry = false;
50              try {
51                  return target.call();
52              }
53              catch (Exception e) { // NOSONAR
54                  // handle exceptions
55                  Throwable t = AludraTestUtil.unwrapInvocationTargetException(e);
56                  recentException = t;
57                  if (toleratedThrowable != null && toleratedThrowable.isAssignableFrom(e.getClass()) && retryCount < maxRetries) {
58                      doRetry = true;
59                      retryCount++;
60                  }
61              }
62          }
63          while (doRetry);
64          // the loop has finished without successful execution, so throw the most recent exception
65          throw recentException;
66      }
67  
68  }
