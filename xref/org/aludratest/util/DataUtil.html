47      /** Parses a boolean value in a string ("tru" or "false").
48       *  @param text the text to parse
49       *  @return <code>true</code> or <code>false</code> */
50      public static boolean parseBoolean(String text) {
51          return ParseUtil.parseBoolean(text);
52      }
53  
54      /**
55       * Compares two string arrays
56       * @param expectedStrings an array of the expected strings
57       * @param actualStrings an array of the actual strings
58       * @return a message of the differences or an empty string if the arrays are equal
59       */
60      public static String expectEqualArrays(Object[] expectedStrings, Object[] actualStrings) {
61          // ENHANCE apply ArrayComparator to recognize missing or added elements
62          if (actualStrings == null || expectedStrings == null) {
63              return "No Values found for comparison. ";
64          }
65          if (!Arrays.equals(expectedStrings, actualStrings)) {
66              return "Arrays do not match. Expected " + Arrays.toString(expectedStrings) + ", " +
67                      "but found " + Arrays.toString(actualStrings) + ". ";
68          }
69          return "";
70      }
71  
72      /**
73       * Verifies that each expected string is found in the array of actual strings.
74       * @param expectedStrings
75       * @param actualStrings
76       * @return a String with a diff message, or an empty string if all strings were contained.
77       */
78      public static String containsStrings(String[] expectedStrings, String[] actualStrings) {
79          StringBuilder mismatches = new StringBuilder();
80          StringBuilder actualString = new StringBuilder();
81          String combinedActualString = null;
82          for (int i = 0; i < actualStrings.length; i++) {
83              actualString.append(actualStrings[i]).append(",");
84          }
85          combinedActualString = actualString.toString();
86          for (int i = 0; i < expectedStrings.length; i++) {
87              if (!(combinedActualString.contains(expectedStrings[i]))) {
88                  mismatches.append(expectedStrings[i]).append(" ");
89              }
90          }
91          return mismatches.toString();
92      }
93  
94      public static String containsString(String expectedString, String[] actualStrings) {
95          boolean found = false;
96          String actualStringUpperCase = null;
97          String expectedStringUpperCase = expectedString.toUpperCase();
98          for (int i = 0; i < actualStrings.length; i++) {
99              actualStringUpperCase = actualStrings[i].toUpperCase();
100             if (actualStringUpperCase.equals(expectedStringUpperCase)) {
101                 found = true;
102                 break;
103             }
104         }
105         if (!found) {
106             return expectedString;
107         }
108         return "";
109     }

111     private static final MostRecentUseCache.Factory<String, Document> docFactory = new MostRecentUseCache.Factory<String, Document>() {
112         @Override
113         public Document create(String html) {
114             Tidy tidy = new Tidy();
115             tidy.setQuiet(true);
116             tidy.setShowWarnings(false);
117             tidy.setShowErrors(0);
118             tidy.setXmlOut(false);
119             Document document = tidy.parseDOM(new StringReader(html), null);
120             return document;
121         }
122     };
123 
124     /* Use an MRU cache for HTML -> DOM mappings. If executing several XPaths on same HTML, this increases performance
125      * significantly. */
126     private static final MostRecentUseCache<String, Document> docCache = new MostRecentUseCache<String, Document>(docFactory, 50);
127 
128     public static NodeList evalXPathInHTML(XPathLocator locator, String html) {
129         return evalXPathInHTML(locator.toString(), html);
130     }
131 
132     public static NodeList evalXPathInHTML(String xpath, String html) {
133         try {
134             Document document = docCache.get(html);
135             XPathExpression expression = XPathFactory.newInstance().newXPath().compile(xpath);
136             return (NodeList) expression.evaluate(document, XPathConstants.NODESET);
137         } catch (XPathExpressionException e) {
138             throw new AutomationException("Illegal XPath: " + xpath, e);
139         }
140     }
