40  /**
41   * Exports a {@link RunnerTree} structure as DOT diagram file.
42   * @author Volker Bergmann
43   */
44  public class RunnerTreeDotExporter {
45  
46      private RunnerTreeDotExporter() { }
47  
48      /**
49       * Main method that interprets the first parameter as class name,
50       * parses its internal test suite structure and exports it to a DOT
51       * file named '&lt;classname&gt;.dot'.
52       * @param args the command line parameters
53       * @throws IOException
54       */
55      public static void main(String[] args) throws IOException {
56          // check preconditions
57          if (args.length != 1) {
58              System.out.println("Please provide the name of the test class or test suite"); //NOSONAR
59              System.exit(AludraTestConstants.EXIT_ILLEGAL_ARGUMENT);
60          }
61          // execute
62          String resourceName = args[0];
63  
64          AludraTest aludraTest = AludraTest.startFramework();
65          try {
66              RunnerTreeBuilder builder = aludraTest.getServiceManager().newImplementorInstance(RunnerTreeBuilder.class);
67              RunnerTree tree = builder.buildRunnerTree(AppUtil.classForPotentialResourceName(resourceName));
68              RunnerTreeUtil.debugSubTree(tree.getRoot(), LoggerFactory.getLogger(RunnerTreeDotExporter.class), "");
69              export(tree, resourceName + ".dot");
70          }
71          finally {
72              aludraTest.stopFramework();
73          }
74      }
75  
76      /**
77       * Exports a {@link RunnerTree} structure as DOT diagram file.
78       * @param tree the tree structure to export
79       * @param fileName the name of the file to create
80       * @throws FileNotFoundException if the file could not be vreated on the file system
81       */
82      public static void export(RunnerTree tree, String fileName) throws FileNotFoundException {
83          DotGraph graph = createGraph(tree);
84          DefaultDotGraphModel model = new DefaultDotGraphModel(graph);
85          DotWriter.persist(model, new FileOutputStream(fileName), Encodings.UTF_8);
86      }
87  
88  
89      // private helper methods --------------------------------------------------
90  
91      private static DotGraph createGraph(RunnerTree tree) {
92          DotGraph graph = DotGraph.newDirectedGraph("nodes").withRankDir(RankDir.LR).withNodeShape(NodeShape.record);
93          addRunnerGroup(tree.getRoot(), graph);
94          return graph;
95      }
96  
97      private static DotNode addRunnerGroup(RunnerGroup runnerGroup, DotGraph graph) {
98          DotNode dotGroup = graph.newNode("\"" + runnerGroup.getName() + "\"");
99          DotNode connectionPoint = dotGroup;
100         // iterate all children
101         for (RunnerNode child : runnerGroup.getChildren()) {
102             // create child node
103             DotNode dotChild;
104             if (child instanceof RunnerGroup) {
105                 dotChild = addRunnerGroup((RunnerGroup) child, graph);
106             } else {
107                 dotChild = addRunnerLeaf((RunnerLeaf) child, graph);
108             }
109             // create edge to new node
110             if (runnerGroup.isParallel() || connectionPoint == dotGroup) { //NOSONAR
111                 dotGroup.newEdgeTo(dotChild).withArrowTail(ArrowShape.ediamond).withArrowHead(ArrowShape.open);
112             } else {
113                 connectionPoint.newEdgeTo(dotChild).withArrowHead(ArrowShape.open);
114             }
115             if (!runnerGroup.isParallel()) {
116                 connectionPoint = dotChild;
117             }
118         }
119         return dotGroup;
120     }
121 
122     private static DotNode addRunnerLeaf(RunnerLeaf leaf, DotGraph graph) {
123         DotNode node = graph.newNode("\"" + leaf.getName() + "\"");
124         return node;
125     }
