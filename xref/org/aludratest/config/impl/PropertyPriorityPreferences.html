32  /**
33   * Preferences wrapper dealing with the complex AludraText configuration overwrite mechanisms.
34   * 
35   * @author falbrech
36   * 
37   */
38  public class PropertyPriorityPreferences extends AbstractPreferences {
39  
40      private static final Logger LOG = LoggerFactory.getLogger(PropertyPriorityPreferences.class);
41  
42      private static final String SYSPROP_PREFIX = "ALUDRATEST_CONFIG/";
43  
44      /**
45       * The pattern for a environment-global service system property (valid for all service instances, until overwritten by an
46       * instance specific property for the same service type). The parameters are:
47       * <ol>
48       * <li>Component / Service ID</li>
49       * <li>Property name (maybe including slashes)</li>
50       * </ol>
51       */
52      private static final String SYSPROP_ENV_NAME = SYSPROP_PREFIX + "{0}/{1}";
53  
54      private static final String SYSPROP_ENV_INSTANCE_NAME = SYSPROP_PREFIX + "{0}/_{1}/{2}";
55  
56      /**
57       * The patterns for property files to search for property to find. They are checked in the order of their appearance in this
58       * array. The parameters passed to <code>MessageFormat.format()</code> are, for each array entry:
59       * <ol>
60       * <li>Environment name</li>
61       * <li>Component / Service ID</li>
62       * <li>Service instance name, or XYZ if no service instance</li>
63       * </ol>
64       */
65      private static final String[] PROPFILE_NAMES = { "config/_{0}/{2}/{1}.properties", "config/_{0}/{1}.properties",
66          "config/{2}/{1}.properties", "config/{1}.properties"
67      };
68  
69      private String componentName;
70  
71      private String serviceInstanceName;
72  
73      private Preferences delegate;
74  
75      private String relativePath;
76  
77  
78      public PropertyPriorityPreferences(String componentName, Preferences delegate) {
79          this(componentName, null, delegate);
80      }
81  
82      public PropertyPriorityPreferences(String componentName, String serviceInstanceName,
83              Preferences delegate) {
84          if (delegate == null) {
85              throw new IllegalArgumentException("delegate is null");
86          }
87          if (componentName == null) {
88              throw new IllegalArgumentException("componentName is null");
89          }
90          this.componentName = componentName;
91          this.serviceInstanceName = serviceInstanceName;
92          this.delegate = delegate;
93      }
94  
95      private PropertyPriorityPreferences(String componentName, String serviceInstanceName,
96              String relativePath,
97              Preferences delegate) {
98          this(componentName, serviceInstanceName, delegate);
99          this.relativePath = relativePath;
100     }
101 
102     @Override
103     public String[] getKeyNames() {
104         // this is quite complex, because all potential files and even the system properties
105         // have to be scanned for potential matches!
106         List<String> keyNames = new ArrayList<String>();
107         keyNames.addAll(Arrays.asList(delegate.getKeyNames()));
108 
109         // properties files on classpath
110         keyNames.addAll(getPropertiesFilesSubNames(false));
111 
112         // system property matches
113         keyNames.addAll(getSystemPropertiesSubNames(false));
114 
115         return keyNames.toArray(new String[0]);
116     }
117 
118     @Override
119     protected String internalGetStringValue(String key) {
120         String fullPropName = (relativePath == null ? key : (relativePath + "/" + key));
121 
122         // top priority - system property
123         String sysPropValue = checkSystemProperties(fullPropName);
124         if (sysPropValue != null) {
125             return sysPropValue;
126         }
127 
128         // now for the .properties files on the classpath
129         String propertiesFilesValue = checkPropertiesFiles(fullPropName);
130         if (propertiesFilesValue != null) {
131             return propertiesFilesValue;
132         }
133 
134         // last fallback: The delegate, which should be default value
135         return delegate.getStringValue(key);
136     }
137 
138     private String checkSystemProperties(String fullPropName) {
139         // check service instance name system property, if service instance is set
140         if (serviceInstanceName != null) {
141             String propName = MessageFormat.format(SYSPROP_ENV_INSTANCE_NAME, componentName, serviceInstanceName,
142                     fullPropName);
143             String value = System.getProperty(propName);
144             if (value != null) {
145                 return value;
146             }
147         }
148         // check component system property
149         String propName = MessageFormat.format(SYSPROP_ENV_NAME, componentName, fullPropName);
150         return System.getProperty(propName);
151     }
152 
153     private List<Properties> loadPropertiesFiles() {
154         List<Properties> result = new ArrayList<Properties>();
155 
156         String instanceName = serviceInstanceName == null ? "XYZ" : serviceInstanceName;
157 
158         ClassLoader cl = Thread.currentThread().getContextClassLoader();
159         if (cl == null) {
160             cl = PropertyPriorityPreferences.class.getClassLoader();
161         }
162 
163         for (String pnPattern : PROPFILE_NAMES) {
164             String pfName = MessageFormat.format(pnPattern, AludraTest.getEnvironmentName(), componentName,
165                     instanceName);
166             URL url = cl.getResource(pfName);
167             if (url != null) {
168                 InputStream in = null;
169                 try {
170                     in = url.openStream();
171                     Properties p = new Properties();
172                     p.load(in);
173                     result.add(p);
174                 }
175                 catch (IOException e) {
176                     LOG.error("Could not load configuration file " + url, e);
177                 }
178                 finally {
179                     try {
180                         if (in != null) {
181                             in.close();
182                         }
183                     }
184                     catch (Exception e) {
185                         LOG.debug("Exception when closing input stream", e);
186                     }
187                 }
188             }
189         }
190 
191         return result;
192     }
193 
194     private String checkPropertiesFiles(String fullPropName) {
195         for (Properties p : loadPropertiesFiles()) {
196             if (p.containsKey(fullPropName)) {
197                 return p.getProperty(fullPropName);
198             }
199         }
200         return null;
201     }
202 
203     private List<String> getPropertiesFilesSubNames(boolean nodeNames) {
204         List<String> result = new ArrayList<String>();
205 
206         for (Properties p : loadPropertiesFiles()) {
207             addMatchingSubNames(p, nodeNames, result);
208         }
209 
210         return result;
211     }
212 
213     private void addMatchingSubNames(Properties p, boolean nodeNames, List<String> result) {
214         for (String key : p.stringPropertyNames()) {
215             if (relativePath == null || key.startsWith(relativePath + "/")) {
216                 String remainderKey = relativePath == null ? key : key.substring(relativePath.length() + 1);
217                 if (!remainderKey.contains("/") && !nodeNames) {
218                     result.add(remainderKey);
219                 }
220                 else if (remainderKey.contains("/") && nodeNames) {
221                     result.add(remainderKey.substring(0, remainderKey.indexOf('/')));
222                 }
223             }
224         }
225     }
226 
227     private List<String> getSystemPropertiesSubNames(boolean nodeNames) {
228         // build temporary property set with all nodes applying to this preferences
229 
230         Properties p = new Properties();
231         for (String key : System.getProperties().stringPropertyNames()) {
232             if (key.startsWith(SYSPROP_PREFIX)) {
233                 // remove prefix, check if applicable to us
234                 key = key.substring(SYSPROP_PREFIX.length());
235                 if (key.startsWith(componentName + "/")) {
236                     key = key.substring(componentName.length() + 1);
237                     // also instance to remove?
238                     if (key.startsWith("_") && key.contains("/")) {
239                         key = key.substring(key.indexOf('/') + 1);
240                     }
241                     p.setProperty(key, "1");
242                 }
243             }
244         }
245 
246         List<String> result = new ArrayList<String>();
247         addMatchingSubNames(p, nodeNames, result);
248         return result;
249     }

251     @Override
252     public Preferences getChildNode(String name) {
253         Preferences child = delegate.getChildNode(name);
254         // could be a path only available via properties files etc, so assume an empty delegate
255         if (child == null) {
256             child = new SimplePreferences();
257         }
258         String path = relativePath == null ? name : relativePath + "/" + name;
259         return new PropertyPriorityPreferences(componentName, serviceInstanceName, path, child);
260     }
261 
262     @Override
263     public String[] getChildNodeNames() {
264         List<String> nodeNames = new ArrayList<String>();
265 
266         // this is quite complex, because all potential files and even the system properties
267         // have to be scanned for potential matches!
268         nodeNames.addAll(Arrays.asList(delegate.getChildNodeNames()));
269 
270         // properties files on classpath
271         nodeNames.addAll(getPropertiesFilesSubNames(true));
272 
273         // system property matches
274         nodeNames.addAll(getSystemPropertiesSubNames(true));
275 
276         return nodeNames.toArray(new String[0]);
277     }
