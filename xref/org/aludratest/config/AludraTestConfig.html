16  package org.aludratest.config;
17  
18  /** Provides the central configuration of AludraTest. <br>
19   * If you need access to an instance of this interface, the preferred way is to ask your context object for a
20   * <code>newComponentInstance()</code> using this Interface class as the key.
21   * 
22   * @author Volker Bergmann
23   * @author falbrech */
24  @InternalComponent(singleton = true)
25  @ConfigProperties({
26      @ConfigProperty(name = AludraTestConfig.NUMBER_OF_THREADS_PROP, type = int.class, description = "The number of threads to use for parallel test case execution", defaultValue = "1"),
27      @ConfigProperty(name = AludraTestConfig.STOP_ON_VERIFICATION_PROP, type = boolean.class, description = "If set to true, test execution stops if an exception is thrown on verfication actions.", defaultValue = "true"),
28      @ConfigProperty(name = AludraTestConfig.STOP_ON_INTERACTION_PROP, type = boolean.class, description = "If set to true, test execution stops if an exception is thrown on interaction actions.", defaultValue = "true"),
29      @ConfigProperty(name = AludraTestConfig.XLS_ROOT_PROP, type = String.class, description = "The root directory containing Excel files with test data", defaultValue = "./src/main/resources/testdata"),
30      @ConfigProperty(name = AludraTestConfig.DEFAULT_LOCALE_PROP, type = String.class, description = "The locale to use as default locale. If not specified, the system default locale is used."),
31      @ConfigProperty(name = AludraTestConfig.CONFIG_TAB_REQUIRED_PROP, type = boolean.class, description = "If set to true, a configuration tab is required in Excel files, and its absence will cause an exception.", defaultValue = "false"),
32      @ConfigProperty(name = AludraTestConfig.IGNORE_ENABLED_PROP, type = boolean.class, description = "If set to true, test case classes can be marked as ignored. Otherwise, the @Ignored annotation will itself be ignored, causing the marked tests to be executed.", defaultValue = "false"),
33      @ConfigProperty(name = AludraTestConfig.NUMERIC_TOLERANCE_PROP, type = double.class, description = "The maximum allowed difference of a current and an expected value. This is for handling rounding issues when dealing with double precision values.", defaultValue = " 0.00000001") })
34  public interface AludraTestConfig extends Configurable {
35  
36      /** Configuration property name. */
37      public static final String NUMBER_OF_THREADS_PROP = "number.of.threads";
38  
39      /** Configuration property name. */
40      public static final String STOP_ON_VERIFICATION_PROP = "stop.testcase.on.verification.exception";
41  
42      /** Configuration property name. */
43      public static final String STOP_ON_INTERACTION_PROP = "stop.testcase.on.interaction.exception";
44  
45      /** Configuration property name. */
46      public static final String XLS_ROOT_PROP = "javatest.xls.root";
47  
48      /** Configuration property name. */
49      public static final String DEFAULT_LOCALE_PROP = "default.locale";
50  
51      /** Configuration property name. */
52      public static final String CONFIG_TAB_REQUIRED_PROP = "config.tab.required";
53  
54      /** Configuration property name. */
55      public static final String IGNORE_ENABLED_PROP = "ignore.enabled";
56  
57      /** Configuration property name. */
58      public static final String NUMERIC_TOLERANCE_PROP = "verification.numeric.difference";
59  
60      // interface ---------------------------------------------------------------
61  
62      /** @return The version of AludraTest, e.g. <code>2.7.0-17</code>. */
63      public String getVersion();
64  
65      /** @return whether to stop test step execution on exceptions in a {@link org.aludratest.service.Interaction} */
66      public boolean isStopTestCaseOnInteractionException();
67  
68      /** @return whether to stop test step execution on exceptions in a {@link org.aludratest.service.Verification} */
69      public boolean isStopTestCaseOnVerificationException();
70  
71      /** @return whether to stop test step execution on exceptions in a location other than {@link org.aludratest.service.Interaction}
72       *         or {@link org.aludratest.service.Verification} */
73      public boolean isStopTestCaseOnOtherException();
74  
75      /** @return the root path to check for Excel files which are consumed to fill test case parameters. */
76      public String getXlsRootPath();
77  
78      /** @return the number of Threads to use by the AludraTest test runner. Default value is 1. */
79      public int getNumberOfThreads();
80  
81      /** @return if each Excel sheet must have a 'config' tab */
82      public boolean isConfigTabRequired();
83  
84      /** @return Indicates if test case classes can be marked as ignored. If <code>false</code>, the Ignored annotation will itself
85       *         be ignored, causing the marked tests to be executed. */
86      public boolean isIgnoreEnabled();
87  
88      /** Returns the numeric tolerance to use for double precision based operations.
89       * 
90       * @return The numeric tolerance to use for double precision based operations. */
91      public double getNumericTolerance();
92  
93  }
