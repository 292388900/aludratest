16  package org.aludratest.impl.log4testing.data;
17  
18  import java.util.ArrayList;
19  import java.util.List;
20  
21  import org.aludratest.testcase.TestStatus;
22  import org.joda.time.DateTime;
23  
24  /**
25   * Represents a Group of TestSteps which could be identified by the name.
26   *  
27   * @author Marcel Malitz
28   * @author Volker Bergmann
29   */
30  public class TestStepGroup extends TestStepContainer {
31  
32      /**
33       * Array which contains all TestSteps
34       */
35      private List<TestStepLog> testSteps = new ArrayList<TestStepLog>();
36      private TestCaseLog parent;
37  
38      /** Constructor.
39       *  @param name the name of the test step group 
40       *  @param parent the parent {@link TestCaseLog} */
41      protected TestStepGroup(String name, TestCaseLog parent) {
42          super(name);
43          this.parent = parent;
44      }
45  
46      /** @return the {@link #parent} */
47      public TestCaseLog getParent() {
48          return parent;
49      }
50  
51      /** @return the number of TestSteps */
52      @Override
53      public int getNumberOfTestSteps() {
54          return testSteps.size();
55      }
56  
57      /** @return the number of Failed TestSteps  */
58      public int getNumberOfFailedTestSteps() {
59          int numberOfFailed = 0;
60          for (TestStepLog testStep : testSteps) {
61              if (testStep.isFailed()) {
62                  ++numberOfFailed;
63              }
64          }
65          return numberOfFailed;
66      }
67  
68      /**
69       * Adds a TestStep to the Array of TestSteps
70       * @return a reference to the TestStepGroup itself
71       */
72      public TestStepLog newTestStep() {
73          TestStepLog testStep = new TestStepLog(this);
74          this.testSteps.add(testStep);
75          return testStep;
76      }
77  
78      /**
79       * Creates a new test step and adds it to this group.
80       * @param status
81       * @param command
82       * @param comment
83       * @return the new test step
84       */
85      public TestStepLog newTestStep(TestStatus status, String command, String comment) {
86          TestStepLog testStep = newTestStep();
87          testStep.setCommand(command);
88          testStep.setComment(comment);
89          testStep.setStatus(status);
90          return testStep;
91      }
92  
93      public void newTestStep(TestStepLog testStep) {
94          this.testSteps.add(testStep);
95          testStep.start();
96          testStep.finish();
97      }
98  
99      /** @return the {@link #testSteps}  */
100     public List<TestStepLog> getTestSteps() {
101         return testSteps;
102     }
103 
104     /** @return the last failed step of the group. */
105     public TestStepLog getLastFailed() {
106         for (int i = testSteps.size() - 1; i >= 0; i--) {
107             TestStepLog step = testSteps.get(i);
108             if (step.isFailed()) {
109                 return step;
110             }
111         }
112         return null;
113     }
114 
115     /** Returns the first TestStep 
116      *  @return the first TestStep*/
117     public TestStepLog getFirstTestStep() {
118         return getTestStep(0);
119     }
120 
121     /** Returns the last TestStep 
122      *  @return the last TestStep */
123     public TestStepLog getLastTestStep() {
124         return getTestStep(testSteps.size() - 1);
125     }
126 
127     /**
128      * Returns the TestStep at the given position
129      * @param index the position of the TestStep
130      * @return TestStep at the Position or <code> null </code> if not found
131      */
132     public TestStepLog getTestStep(int index) {
133         if (testSteps.size() > index && index >= 0) {
134             return testSteps.get(index);
135         } else {
136             return null;
137         }
138     }
139 
140     @Override
141     public DateTime getStartingTime() {
142         if (testSteps.size() > 0) {
143             return testSteps.get(0).getStartingTime();
144         } else {
145             return null;
146         }
147     }
148 
149     @Override
150     public DateTime getFinishingTime() {
151         if (testSteps.size() > 0) {
152             return testSteps.get(testSteps.size() - 1).getFinishingTime();
153         } else {
154             return null;
155         }
156     }
157 
158     @Override
159     public TestStatus getStatus() {
160         TestStepLog lastFailed = getLastFailed();
161         if (lastFailed != null) {
162             return lastFailed.getStatus();
163         } else if (testSteps.size() == 0) {
164             return TestStatus.PASSED;
165         } else if (testSteps.get(0).getStatus().equals(TestStatus.IGNORED)) {
166             return TestStatus.IGNORED;
167         } else {
168             return TestStatus.PASSED;
169         }
170     }
171 
172 }
