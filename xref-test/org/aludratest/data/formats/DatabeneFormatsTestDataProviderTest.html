47      private DatabeneFormatsTestDataProvider createProvider() throws Exception {
48          DatabeneFormatsTestDataProvider provider = new DatabeneFormatsTestDataProvider();
49          AludraTestConfigImpl config = new AludraTestConfigImpl();
50          DataConfigurationImpl dataConfig = new DataConfigurationImpl();
51          DefaultConfigurator configurator = new DefaultConfigurator();
52          configurator.configure(config);
53          configurator.configure(dataConfig);
54  
55          ReflectionUtils.setVariableValueInObject(provider, "aludraConfig", config);
56          ReflectionUtils.setVariableValueInObject(provider, "dataConfig", dataConfig);
57          return provider;
58      }
59  
60      @Test
61      public void testNoArgMethod() throws Exception {
62          Method noArgMethod = BeanUtil.getMethod(DatabeneFormatsTestDataProviderTest.class, "noArgMethod");
63          List<TestCaseData> testDataSets = createProvider().getTestDataSets(noArgMethod);
64          assertTrue("No test data sets created for no-arg method", testDataSets.size() >= 1);
65          assertTrue("Unexpected test data sets for no-arg method", testDataSets.size() <= 1);
66          assertTrue("Unexpected parameters loaded for no-arg method", testDataSets.get(0).getData().length == 0);
67      }
68  
69      @Test
70      public void testSingleArgMethod() throws Exception {
71          Method singleArgMethod = BeanUtil.getMethod(DatabeneFormatsTestDataProviderTest.class, "singleArgMethod", PersonBean.class);
72          List<TestCaseData> testDataSets = createProvider().getTestDataSets(singleArgMethod);
73          assertTrue("No test data sets created for single-arg method", testDataSets.size() >= 1);
74          assertEquals(2, testDataSets.size());
75          assertPerson("Alice", 23, "London", "Dover", testDataSets.get(0).getData());
76          assertPerson("Bob", 34, "New York", "Hauppauge", testDataSets.get(1).getData());
77      }
78  
79      @Test
80      public void testSingleArgMethodWithOffset() throws Exception {
81          Method singleArgMethod = BeanUtil.getMethod(DatabeneFormatsTestDataProviderTest.class, "singleArgMethodWithOffset", PersonBean.class);
82          List<TestCaseData> testDataSets = createProvider().getTestDataSets(singleArgMethod);
83          assertTrue("No test data sets created for single-arg method", testDataSets.size() >= 1);
84          assertEquals(1, testDataSets.size());
85          assertPerson("Bob", 34, "New York", "Hauppauge", testDataSets.get(0).getData());
86      }
87  
88      @Test
89      public void testMultiArgMethod() throws Exception {
90          Method multiArgMethod = BeanUtil.getMethod(DatabeneFormatsTestDataProviderTest.class, "multiArgMethod", PersonBean.class, AddressBean.class);
91          List<TestCaseData> testDataSets = createProvider().getTestDataSets(multiArgMethod);
92          assertTrue("No test data sets created for multi-arg method", testDataSets.size() >= 1);
93          assertEquals(2, testDataSets.size());
94          assertPersonAndAddress("Max", 23, "Munich", testDataSets.get(0).getData());
95          assertPersonAndAddress("Jens", 34, "Hamburg", testDataSets.get(1).getData());
96      }
97  
98      @Test
99      public void testEmptyValuesMethod() throws Exception {
100         Method nullValuesMethod = BeanUtil.getMethod(DatabeneFormatsTestDataProviderTest.class, "withNullValuesMethod",
101                 PersonBean.class);
102         List<TestCaseData> testDataSets = createProvider().getTestDataSets(nullValuesMethod);
103         assertTrue("No test data sets created for null values method", testDataSets.size() >= 1);
104         assertEquals(5, testDataSets.size());
105 
106         assertPerson("Alice", 23, null, null, testDataSets.get(0).getData());
107         assertPerson("Bob", 34, null, null, testDataSets.get(1).getData());
108         assertPerson(null, 35, null, null, testDataSets.get(2).getData());
109         assertPerson("", 36, null, null, testDataSets.get(3).getData());
110         assertPerson(null, 37, null, null, testDataSets.get(4).getData());
111     }

114     public void testNullHeaderMethod() throws Exception {
115         Method nullValuesMethod = BeanUtil.getMethod(DatabeneFormatsTestDataProviderTest.class, "nullHeaderMethod",
116                 PersonBean.class);
117         List<TestCaseData> testDataSets = createProvider().getTestDataSets(nullValuesMethod);
118         assertTrue("No test data sets created for null values method", testDataSets.size() >= 1);

121     // private helper methods --------------------------------------------------
122 
123     private void assertPerson(String name, int age, String city1, String city2, Data[] args) {
124         assertEquals(1, args.length);
125         PersonBean person = (PersonBean) args[0];
126         assertEquals(name, person.getName());
127         assertEquals(age, person.getAge());
128         if (city1 != null) {
129             assertNotNull(person.getAddresses());
130             assertTrue(person.getAddresses().size() > 0);
131             assertEquals(city1, person.getAddresses().get(0).getCity());
132             if (city2 != null) {
133                 assertTrue(person.getAddresses().size() > 1);
134                 assertEquals(city2, person.getAddresses().get(1).getCity());
135             }
136         }
137     }
138 
139     private void assertPersonAndAddress(String name, int age, String city, Data[] args) {
140         assertEquals(2, args.length);
141         Data[] personArray = new Data[] { args[0] };
142         assertPerson(name, age, null, null, personArray);
143         AddressBean address = (AddressBean) args[1];
144         assertEquals(city, address.getCity());
145     }
