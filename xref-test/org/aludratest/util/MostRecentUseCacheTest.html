32      @Test
33      public void testHashCode() {
34          MostRecentUseCache<Helper1, String> cache = new MostRecentUseCache<Helper1, String>(helperFactory, 10);
35  
36          Helper1 key1 = new Helper1("Test1");
37          Helper1 key2 = new Helper1("Test2");
38          assertEquals("Test1", cache.get(key1));
39          assertEquals("Test1", cache.get(key2)); // because same hash code and value exists
40          assertEquals(1, cache.getSize());
41  
42          cache.clear();
43          assertEquals("Test2", cache.get(key2));
44          assertEquals("Test2", cache.get(key1));
45          assertEquals(1, cache.getSize());
46  
47          // this also tests the reuse of objects, so no special test for reusing objects.
48      }

50      @Test
51      public void testMaxSize() {
52          CountingFactory factory = new CountingFactory();
53          MostRecentUseCache<String, String> cache = new MostRecentUseCache<String, String>(factory, 3);
54          cache.get("Test1");
55          cache.get("Test2");
56          cache.get("Test3");
57          assertEquals(1, factory.createCounter.get("Test1").intValue());
58          assertEquals(1, factory.createCounter.get("Test2").intValue());
59          assertEquals(1, factory.createCounter.get("Test3").intValue());
60          assertEquals(3, cache.getSize());
61          cache.get("Test4");
62          assertEquals(1, factory.createCounter.get("Test4").intValue());
63          assertEquals(3, cache.getSize());
64          cache.get("Test1");
65          cache.get("Test3");
66          cache.get("Test4");
67          assertEquals(3, cache.getSize());
68          assertEquals(2, factory.createCounter.get("Test1").intValue());
69          assertEquals(1, factory.createCounter.get("Test2").intValue());
70          assertEquals(1, factory.createCounter.get("Test3").intValue());
71          assertEquals(1, factory.createCounter.get("Test4").intValue());
72      }

74      @Test
75      public void testHashCalculator() {
76          HashCalculator<Helper1> calculator = new HashCalculator<MostRecentUseCacheTest.Helper1>() {
77              @Override
78              public String hash(Helper1 object) {
79                  return object.value; // FULL "HASH"
80              }
81          };
82          MostRecentUseCache<Helper1, String> cache = new MostRecentUseCache<Helper1, String>(helperFactory, 10, calculator);
83  
84          Helper1 key1 = new Helper1("Test1");
85          Helper1 key2 = new Helper1("Test2");
86          assertEquals("Test1", cache.get(key1));
87          assertEquals("Test2", cache.get(key2));
88          assertEquals(2, cache.getSize());
89  
90          cache.clear();
91          assertEquals("Test2", cache.get(key2));
92          assertEquals("Test1", cache.get(key1));
93          assertEquals(2, cache.getSize());
94      }

96      @Test
97      public void testThreadSafety() {
98          CountingFactory factory = new CountingFactory();
99          final MostRecentUseCache<String, String> cache = new MostRecentUseCache<String, String>(factory, 10);
100         Runnable runGet = new Runnable() {
101             @Override
102             public void run() {
103                 try {
104                     // give all threads the chance to start parallel
105                     Thread.sleep(100);
106                 }
107                 catch (InterruptedException e) {
108                     return;
109                 }
110                 cache.get("Test1");
111                 cache.get("Test2");
112                 cache.get("Test3");
113                 cache.get("Test4");
114             }
115         };
116 
117         ExecutorService svc = Executors.newFixedThreadPool(10);
118         for (int i = 0; i < 10; i++) {
119             svc.execute(runGet);
120         }
121         svc.shutdown();
122         try {
123             svc.awaitTermination(2000, TimeUnit.MILLISECONDS);
124         }
125         catch (InterruptedException e) {
126             return;
127         }
128 
129         assertEquals(1, factory.createCounter.get("Test1").intValue());
130         assertEquals(1, factory.createCounter.get("Test2").intValue());
131         assertEquals(1, factory.createCounter.get("Test3").intValue());
132         assertEquals(1, factory.createCounter.get("Test4").intValue());
133     }

135     private static class Helper1 {
136         private String value;
137 
138         public Helper1(String value) {
139             this.value = value;
140         }
141 
142         @Override
143         public int hashCode() {
144             return 1;
145         }
146     }
147 
148     private static final Factory<Helper1, String> helperFactory = new MostRecentUseCache.Factory<Helper1, String>() {
149         @Override
150         public String create(Helper1 key) {
151             return key.value;
152         }
153     };
154 
155     private static class CountingFactory implements Factory<String, String> {
156 
157         private Map<String, Integer> createCounter = new HashMap<String, Integer>();
158 
159         @Override
160         public String create(String key) {
161             Integer i = createCounter.get(key);
162             i = (i == null ? 1 : (i.intValue() + 1));
163             createCounter.put(key, i);
164 
165             return key;
166         }
167     }
