35  /**
36   * Tests the {@link TestSuiteLog}'s state management and event handling for thread safety.
37   * @author Volker Bergmann
38   */
39  @SuppressWarnings("javadoc")
40  public class MultiThreadedTestCaseTest {
41  
42      @Before
43      public void clear() {
44          TestLogger.clear();
45          ObserverTestUtil.removeAllTestObservers();
46          TestObserverManager.getInstance().addObserver(new PendingTestsObserver());
47      }
48  
49      @Test
50      public void testFlatStructureWithSlowTests() throws Exception {
51          performTest(0, 0, 5, 500);
52      }
53  
54      @Test
55      public void testNested() throws Exception {
56          performTest(2, 3, 3, 100);
57      }
58  
59      @Test
60      public void testComplexHighlyConcurrent() throws Exception {
61          performTest(3, 4, 20, 0);
62      }
63  
64      private void performTest(int hierarchyDepth, int childSuiteCount, int childCaseCount, int sleepTime) throws InterruptedException {
65          TestSuiteLog root = new TestSuiteLog("root");
66          List<TestCaseLog> testCases = createTestCases(root, hierarchyDepth, childSuiteCount, childCaseCount, new ArrayList<TestCaseLog>());
67          SuitePrinter.print(root);
68          List<Thread> threads = createThreads(testCases, sleepTime);
69          runThreadsAndWait(threads);
70          for (TestCaseLog testCase : testCases) {
71              assertTrue(testCase.isFinished());
72              assertFalse(testCase.getLastTestStep().getErrorMessage(), testCase.isFailed());
73          }
74      }
75  
76      private List<TestCaseLog> createTestCases(TestSuiteLog parent, int depth, int childSuiteCount, int childCaseCount, List<TestCaseLog> testCases) {
77          for (int i = 0; i < childCaseCount; i++) {
78              TestCaseLog childCase = TestLogger.getTestCase(parent.getName() + "-case_" + (i + 1));
79              parent.add(childCase);
80              testCases.add(childCase);
81          }
82          for (int i = 0; i < childSuiteCount; i++) {
83              TestSuiteLog childSuite = TestLogger.getTestSuite(parent.getName() + "-suite_" + (i + 1));
84              parent.add(childSuite);
85              if (depth > 0)
86                  createTestCases(childSuite, depth - 1, childSuiteCount, childCaseCount, testCases);
87          }
88          return testCases;
89      }
90  
91      private List<Thread> createThreads(List<TestCaseLog> testCases, int sleepTime) {
92          List<Thread> threads = new ArrayList<Thread>(testCases.size());
93          for (TestCaseLog testCase : testCases) {
94              threads.add(new TestCaseThread(testCase, sleepTime));
95          }
96          return threads;
97      }
98  
99      private void runThreadsAndWait(List<Thread> threads) throws InterruptedException {
100         for (Thread thread : threads) {
101             thread.start();
102         }
103         for (Thread thread : threads) {
104             thread.join();
105         }
106     }
107 
108     static class TestCaseThread extends Thread implements Thread.UncaughtExceptionHandler {
109 
110         private TestCaseLog testCase;
111         private long sleepTime;
112 
113         private TestCaseThread(TestCaseLog testCase, long sleepTime) {
114             this.testCase = testCase;
115             this.sleepTime = sleepTime;
116             setUncaughtExceptionHandler(this);
117         }
118 
119         @Override
120         public void run() {
121             try {
122                 //System.out.println("Starting " + testCase);
123                 testCase.start();
124                 testCase.newTestStepGroup("g1");
125                 testCase.newTestStep();
126                 Thread.sleep(sleepTime);
127                 testCase.finish();
128                 //System.out.println("Finished " + testCase);
129             } catch (InterruptedException e) {
130                 e.printStackTrace();
131             }
132         }
133 
134         public void uncaughtException(Thread thread, Throwable throwable) {
135             TestStepLog lastStep = testCase.getLastTestStep();
136             lastStep.setStatus(TestStatus.FAILEDAUTOMATION);
137             lastStep.setErrorMessage("Exception occurred: " + throwable.toString());
138         }
139     }
